74fn dfs(edges: &BTreeMap<String, BTreeSet<String>>, ty: &String, set: &mut BTreeSet<String>) {
75    if let Some(subtypes) = edges.get(ty) {
76        for st in subtypes.iter() {
77            set.insert(st.clone());
78            dfs(edges, st, set);
79        }
80    }
81}

83fn check_function_protocol_is_in_range(func: &Function) -> bool {
84    let min_supported_proto_is_too_new = func
85        .min_supported_protocol
86        .is_some_and(|v| v > LEDGER_PROTOCOL_VERSION);
87    let max_supported_proto_is_too_old = func
88        .max_supported_protocol
89        .is_some_and(|v| v < LEDGER_PROTOCOL_VERSION);
90    !min_supported_proto_is_too_new && !max_supported_proto_is_too_old
91}

158pub fn generate_wasm_module_calling_host_functions(file_lit: LitStr) -> Result<TokenStream, Error> {
159    let root: crate::Root = crate::load_env_file(file_lit)?;
160    let impls = root
161        .modules
162        .iter()
163        .flat_map(|m| {
164            m.functions
165                .clone()
166                .into_iter()
167                .map(move |f| (m.export.clone(), f))
168        })
169        .map(|(mod_export, hf)| {
170            let fn_export = hf.export;
171            let arity = hf.args.len() as u32;
172            let wasm_module = format_ident!("wasm_module_calling_{}", hf.name);
173            quote! {
174                // define the wasms
175                fn #wasm_module() -> Vec<u8> {
176                    let mut me = ModEmitter::default_with_test_protocol();
177                    let f0 = me.import_func(#mod_export, #fn_export, Arity(#arity));
178                    let mut fe = me.func(Arity(#arity), 0);
179                    for i in 0..#arity {
180                        fe.push(Operand::Local(LocalRef(i)));
181                    }
182                    fe.call_func(f0);
183                    fe.drop();
184                    fe.push(Symbol::try_from_small_str("pass").unwrap());
185                    fe.finish_and_export("test").finish()
186                }
187            }
188        });
189
190    Ok(quote! {
191        #(#impls)*
192    })
193}

195pub fn generate_hostfn_call_with_wrong_types(file_lit: LitStr) -> Result<TokenStream, Error> {
196    let root: crate::Root = crate::load_env_file(file_lit)?;
197
198    // We first build the type relationship graph from the hardcoded values.
199    // The goal is to figure out all the ancestors and descentents of a type.
200    let mut edges: BTreeMap<String, BTreeSet<String>> = BTreeMap::new();
201    let mut reverse_edges: BTreeMap<String, BTreeSet<String>> = BTreeMap::new();
202    let mut all_test_candidate_types: BTreeSet<String> = BTreeSet::new();
203    for (ty, subty) in SUBTYPES.iter() {
204        edges
205            .entry(ty.to_string())
206            .or_default()
207            .insert(subty.to_string());
208        reverse_edges
209            .entry(subty.to_string())
210            .or_default()
211            .insert(ty.to_string());
212        all_test_candidate_types.insert(ty.to_string());
213        all_test_candidate_types.insert(subty.to_string());
214    }
215
216    let mut children_of_type: BTreeMap<String, BTreeSet<String>> = BTreeMap::new();
217    for (ty, _) in edges.iter() {
218        let children = children_of_type.entry(ty.clone()).or_default();
219        dfs(&edges, ty, children);
220    }
221
222    let mut parents_of_type: BTreeMap<String, BTreeSet<String>> = BTreeMap::new();
223    for (ty, _) in reverse_edges.iter() {
224        let parents = parents_of_type.entry(ty.clone()).or_default();
225        dfs(&reverse_edges, ty, parents);
226    }
227
228    // Collect all the types appeared as an argument in a host function
229    // and map it to the host function and position it appears in.
230    let mut type_to_fn_arg: BTreeMap<String, Vec<(Function, usize)>> = BTreeMap::new();
231    for m in root.modules.clone() {
232        for f in m.functions.clone() {
233            // checks if the current ledger protocol version falls between
234            // supported protocol versions of this function
235            if check_function_protocol_is_in_range(&f) {
236                for (i, a) in f.args.iter().enumerate() {
237                    type_to_fn_arg
238                        .entry(a.r#type.clone())
239                        .or_default()
240                        .push((f.clone(), i));
241                }
242            }
243        }
244    }
245
246    // Because of the combination of "all-host-fns * all-args-of-a-fn * all-the-wrong-types-of-that-arg" is very large,
247    // we go through each appeared type, and choose one hostfn -- `target_fn` to fun the test with.
248    let test_impls = type_to_fn_arg
249        .iter()
250        .filter(|entry| all_test_candidate_types.contains(entry.0))
251        .map(|entry| {
252            let target_type = entry.0;
253            // we just always pick the first function, instead of a random one
254            let (target_fn, target_arg_pos) = entry.1.first().unwrap();
255            let wasm_module = format_ident!("wasm_module_calling_{}", &target_fn.name);
256            let test_wrong_arg_type =
257                format_ident!("dispatch_with_wrong_arg_type_{}", &target_fn.name);
258            let target_min_vers = target_fn.min_supported_protocol.unwrap_or(0);
259            let target_max_vers = target_fn
260                .max_supported_protocol
261                .unwrap_or(LEDGER_PROTOCOL_VERSION);
262
263            // There are three types of possibilities between two types "target" and "input":
264            // 1. Compatible -- the target type is a parent (say `Val`), then passing in any child type value is fine.
265            // 2. Maybe compatible -- the target value is a child (say `SymbolSmall`), then passing in a parent type (e.g. `Symbol`) maybe compatible
266            // 3. Incompatible -- if the target and input types are neither 1 or 2, e.g `VecObject` and `MapObject`.
267            // For simplicity, our tests cover scenario 3.
268            let calls = all_test_candidate_types
269                .iter()
270                .filter(|&t| {
271                    !(t == target_type
272                        || children_of_type
273                            .get(t)
274                            .is_some_and(|s| s.contains(target_type))
275                        || parents_of_type
276                            .get(t)
277                            .is_some_and(|s| s.contains(target_type)))
278                })
279                .map(|t| {
280                    let input_args = target_fn.args.iter().enumerate().map(|(i, a)| {
281                        if i == *target_arg_pos {
282                            let wrong_ty = format_ident!("{}", t);
283                            quote! {
284                                #wrong_ty::test_val()
285                            }
286                        } else {
287                            let rest_ty = format_ident!("{}", a.r#type);
288                            quote! {
289                                #rest_ty::test_val()
290                            }
291                        }
292                    });
293
294                    quote! {
295                        {
296                            let args = HostVec::from_vec(vec![ #(#input_args),*])?;
297                            let args = host.add_host_object(args)?;
298                            let res = host.call(
299                                contract_id_obj,
300                                Symbol::try_from_small_str("test")?,
301                                args,
302                            );
303                            assert!(HostError::result_matches_err(
304                                res,
305                                (ScErrorType::Value, ScErrorCode::InvalidInput)
306                            ));
307                        }
308                    }
309                });
310
311            quote! {
312                #[test]
313                fn #test_wrong_arg_type() -> Result<(), HostError> {
314                    let wasm = #wasm_module();
315                    let host = observe_host!(Host::test_host_with_recording_footprint());
316                    host.as_budget().reset_unlimited()?;
317                    let proto = host.get_ledger_protocol_version()?;
318                    if proto < #target_min_vers || proto > #target_max_vers {
319                        return Ok(());
320                    }
321                    let contract_id_obj = host.register_test_contract_wasm(wasm.as_slice());
322                    #(#calls)*
323                    Ok(())
324                }
325            }
326        });
327
328    Ok(quote! {
329        #(#test_impls)*
330    })
331}

333pub fn generate_hostfn_call_with_invalid_obj_handles(
334    file_lit: LitStr,
335) -> Result<TokenStream, Error> {
336    let root: crate::Root = crate::load_env_file(file_lit)?;
337
338    let mut all_test_candidate_types: BTreeSet<String> = BTreeSet::new();
339    for (ty, subty) in SUBTYPES.iter() {
340        all_test_candidate_types.insert(ty.to_string());
341        all_test_candidate_types.insert(subty.to_string());
342    }
343    let special_case_fns: BTreeSet<String> = SPECIAL_CASES.iter().map(|s| s.to_string()).collect();
344
345    let test_impls = root
346        .modules
347        .iter()
348        .flat_map(|m| m.functions.clone().into_iter())
349        .filter(check_function_protocol_is_in_range)
350        .flat_map(|f| {
351            f.args
352                .clone()
353                .into_iter()
354                .enumerate()
355                .map(move |(i, a)| ((f.clone(), f.args.clone()), (i, a)))
356        })
357        .filter(|((f, _), (_, arg))| {
358            !special_case_fns.contains(&f.name) && arg.r#type.ends_with("Object")
359        })
360        .map(|((func, args), (pos, _))| {
361            let wasm_module = format_ident!("wasm_module_calling_{}", func.name);
362            let fn_ident = format_ident!("invalid_object_handle_{}_arg_{}", func.name, pos);
363            let target_min_vers = func.min_supported_protocol.unwrap_or(0);
364            let target_max_vers = func
365                .max_supported_protocol
366                .unwrap_or(LEDGER_PROTOCOL_VERSION);
367            let args = args.iter().enumerate().map(|(i, a)| {
368                let ty_ident = format_ident!("{}", a.r#type);
369                // if an arg is Object type, but it is not our test target, we generate an valid handle for it
370                if a.r#type.ends_with("Object") && i != pos {
371                    // we must handle some special cases, e.g. a valid signature must be 64 bytes
372                    if func.name == "verify_sig_ed25519" && i == 2 {
373                        quote! {
374                            #ty_ident::test_object_with_initial_length(&host, 64).to_val()
375                        }
376                    } else {
377                        quote! {
378                            #ty_ident::test_object(&host).to_val()
379                        }
380                    }
381                } else {
382                    quote! {
383                        #ty_ident::test_val()
384                    }
385                }
386            });
387
388            quote! {
389                #[test]
390                fn #fn_ident() -> Result<(), HostError> {
391                    let wasm = #wasm_module();
392                    let host = observe_host!(Host::test_host_with_recording_footprint());
393                    host.as_budget().reset_unlimited()?;
394                    let proto = host.get_ledger_protocol_version()?;
395                    if proto < #target_min_vers || proto > #target_max_vers {
396                        return Ok(());
397                    }
398                    let contract_id_obj = host.register_test_contract_wasm(wasm.as_slice());
399
400                    let args = HostVec::from_vec(vec![ #(#args),*])?;
401                    let args = host.add_host_object(args)?;
402                    let res = host.call(
403                        contract_id_obj,
404                        Symbol::try_from_small_str("test")?,
405                        args,
406                    );
407                    assert!(HostError::result_matches_err(
408                        res,
409                        (ScErrorType::Value, ScErrorCode::InvalidInput)
410                    ));
411
412                    Ok(())
413                }
414            }
415        });
416
417    // manually handle the special case functions
418    let bl_impls = generate_invalid_obj_call_for_special_cases();
419
420    Ok(quote! {
421        #(#test_impls)*
422        #bl_impls
423    })
