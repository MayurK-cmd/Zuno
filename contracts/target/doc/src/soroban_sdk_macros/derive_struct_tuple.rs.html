1use itertools::MultiUnzip;
2use proc_macro2::{Literal, TokenStream as TokenStream2};
3use quote::{format_ident, quote};
4use syn::{Attribute, DataStruct, Error, Ident, Path, Visibility};
5
6use stellar_xdr::curr as stellar_xdr;
7use stellar_xdr::{
8    ScSpecEntry, ScSpecTypeDef, ScSpecUdtStructFieldV0, ScSpecUdtStructV0, StringM, WriteXdr,
9};
10
11use crate::{doc::docs_from_attrs, map_type::map_type, DEFAULT_XDR_RW_LIMITS};
12
13pub fn derive_type_struct_tuple(
14    path: &Path,
15    vis: &Visibility,
16    ident: &Ident,
17    attrs: &[Attribute],
18    data: &DataStruct,
19    spec: bool,
20    lib: &Option<String>,
21) -> TokenStream2 {
22    // Collect errors as they are encountered and emit them at the end.
23    let mut errors = Vec::<Error>::new();
24
25    let fields = &data.fields;
26    let field_count_usize: usize = fields.len();
27
28    let (field_specs, field_idx_lits, try_from_xdrs, try_into_xdrs): (Vec<_>, Vec<_>, Vec<_>, Vec<_>) = fields
29        .iter()
30        .enumerate()
31        .map(|(field_idx, field)| {
32            // For tuple structs that have unnamed fields, use the field index
33            // as the token to reference the field.
34            let field_idx_lit = Literal::usize_unsuffixed(field_idx);
35            let field_name = format!("{}", field_idx);
36            let field_spec = ScSpecUdtStructFieldV0 {
37                doc: docs_from_attrs(&field.attrs),
38                name: field_name.try_into().unwrap_or_else(|_| StringM::default()),
39                type_: match map_type(&field.ty, false) {
40                    Ok(t) => t,
41                    Err(e) => {
42                        errors.push(e);
43                        ScSpecTypeDef::I32
44                    }
45                },
46            };
47            let try_from_xdr = quote! {
48                #field_idx_lit: {
49                    let rv: #path::Val = (&vec[#field_idx_lit].clone()).try_into_val(env).map_err(|_| #path::xdr::Error::Invalid)?;
50                    rv.try_into_val(env).map_err(|_| #path::xdr::Error::Invalid)?
51                }
52            };
53            let try_into_xdr = quote! {
54                (&val.#field_idx_lit).try_into().map_err(|_| #path::xdr::Error::Invalid)?
55            };
56            (field_spec, field_idx_lit, try_from_xdr, try_into_xdr)
57        })
58        .multiunzip();
59
60    // If errors have occurred, render them instead.
61    if !errors.is_empty() {
62        let compile_errors = errors.iter().map(Error::to_compile_error);
63        return quote! { #(#compile_errors)* };
64    }
65
66    // Generated code spec.
67    let spec_gen = if spec {
68        let spec_entry = ScSpecEntry::UdtStructV0(ScSpecUdtStructV0 {
69            doc: docs_from_attrs(attrs),
70            lib: lib.as_deref().unwrap_or_default().try_into().unwrap(),
71            name: ident.to_string().try_into().unwrap(),
72            fields: field_specs.try_into().unwrap(),
73        });
74        let spec_xdr = spec_entry.to_xdr(DEFAULT_XDR_RW_LIMITS).unwrap();
75        let spec_xdr_lit = proc_macro2::Literal::byte_string(spec_xdr.as_slice());
76        let spec_xdr_len = spec_xdr.len();
77        let spec_ident = format_ident!("__SPEC_XDR_TYPE_{}", ident.to_string().to_uppercase());
78        Some(quote! {
79            #[cfg_attr(target_family = "wasm", link_section = "contractspecv0")]
80            pub static #spec_ident: [u8; #spec_xdr_len] = #ident::spec_xdr();
81
82            impl #ident {
83                pub const fn spec_xdr() -> [u8; #spec_xdr_len] {
84                    *#spec_xdr_lit
85                }
86            }
87        })
88    } else {
89        None
90    };
91
92    // Output.
93    let mut output = quote! {
94        #spec_gen
95
96        impl #path::TryFromVal<#path::Env, #path::Val> for #ident {
97            type Error = #path::ConversionError;
98            #[inline(always)]
99            fn try_from_val(env: &#path::Env, val: &#path::Val) -> Result<Self, #path::ConversionError> {
100                use #path::{TryIntoVal,EnvBase,ConversionError,VecObject,Val};
101                let vec: VecObject = (*val).try_into().map_err(|_| ConversionError)?;
102                let mut vals: [Val; #field_count_usize] = [Val::VOID.to_val(); #field_count_usize];
103                env.vec_unpack_to_slice(vec, &mut vals).map_err(|_| ConversionError)?;
104                Ok(Self{
105                    #(#field_idx_lits: vals[#field_idx_lits].try_into_val(env).map_err(|_| ConversionError)?),*
106                })
107            }
108        }
109
110        impl #path::TryFromVal<#path::Env, #ident> for #path::Val {
111            type Error = #path::ConversionError;
112            #[inline(always)]
113            fn try_from_val(env: &#path::Env, val: &#ident) -> Result<Self, #path::ConversionError> {
114                use #path::{TryIntoVal,EnvBase,ConversionError,Val};
115                let vals: [Val; #field_count_usize] = [
116                    #((&val.#field_idx_lits).try_into_val(env).map_err(|_| ConversionError)?),*
117                ];
118                Ok(env.vec_new_from_slice(&vals).map_err(|_| ConversionError)?.into())
119            }
120        }
121    };
122
123    // Additional output when testutils are enabled.
124    if cfg!(feature = "testutils") {
125        let arbitrary_tokens =
126            crate::arbitrary::derive_arbitrary_struct_tuple(path, vis, ident, data);
127        output.extend(quote! {
128            impl #path::TryFromVal<#path::Env, #path::xdr::ScVec> for #ident {
129                type Error = #path::xdr::Error;
130                #[inline(always)]
131                fn try_from_val(env: &#path::Env, val: &#path::xdr::ScVec) -> Result<Self, #path::xdr::Error> {
132                    use #path::xdr::Validate;
133                    use #path::TryIntoVal;
134                    let vec = val;
135                    if vec.len() != #field_count_usize {
136                        return Err(#path::xdr::Error::Invalid);
137                    }
138                    Ok(Self{
139                        #(#try_from_xdrs,)*
140                    })
141                }
142            }
143
144            impl #path::TryFromVal<#path::Env, #path::xdr::ScVal> for #ident {
145                type Error = #path::xdr::Error;
146                #[inline(always)]
147                fn try_from_val(env: &#path::Env, val: &#path::xdr::ScVal) -> Result<Self, #path::xdr::Error> {
148                    if let #path::xdr::ScVal::Vec(Some(vec)) = val {
149                        <_ as #path::TryFromVal<_, _>>::try_from_val(env, vec)
150                    } else {
151                        Err(#path::xdr::Error::Invalid)
152                    }
153                }
154            }
155
156            impl TryFrom<&#ident> for #path::xdr::ScVec {
157                type Error = #path::xdr::Error;
158                #[inline(always)]
159                fn try_from(val: &#ident) -> Result<Self, #path::xdr::Error> {
160                    extern crate alloc;
161                    use #path::TryFromVal;
162                    Ok(#path::xdr::ScVec(alloc::vec![
163                        #(#try_into_xdrs,)*
164                    ].try_into()?))
165                }
166            }
167
168            impl TryFrom<#ident> for #path::xdr::ScVec {
169                type Error = #path::xdr::Error;
170                #[inline(always)]
171                fn try_from(val: #ident) -> Result<Self, #path::xdr::Error> {
172                    (&val).try_into()
173                }
174            }
175
176            impl TryFrom<&#ident> for #path::xdr::ScVal {
177                type Error = #path::xdr::Error;
178                #[inline(always)]
179                fn try_from(val: &#ident) -> Result<Self, #path::xdr::Error> {
180                    Ok(#path::xdr::ScVal::Vec(Some(val.try_into()?)))
181                }
182            }
183
184            impl TryFrom<#ident> for #path::xdr::ScVal {
185                type Error = #path::xdr::Error;
186                #[inline(always)]
187                fn try_from(val: #ident) -> Result<Self, #path::xdr::Error> {
188                    (&val).try_into()
189                }
190            }
191
192            #arbitrary_tokens
193        });
194    }
195    output
196}
