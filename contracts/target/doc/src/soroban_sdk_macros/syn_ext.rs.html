16/// Gets methods from the implementation that have public visibility. For
17/// methods that are inherently implemented this is methods that have a pub
18/// visibility keyword. For methods that are implementing a trait the pub is
19/// assumed and so all methods are returned.
20pub fn impl_pub_methods(imp: &ItemImpl) -> impl Iterator<Item = &ImplItemFn> {
21    imp.items
22        .iter()
23        .filter_map(|i| match i {
24            ImplItem::Fn(m) => Some(m),
25            _ => None,
26        })
27        .filter(|m| imp.trait_.is_some() || matches!(m.vis, Visibility::Public(_)))
28}

30/// Gets methods from the trait.
31pub fn trait_methods(imp: &ItemTrait) -> impl Iterator<Item = &TraitItemFn> {
32    imp.items.iter().filter_map(|i| match i {
33        TraitItem::Fn(m) => Some(m),
34        _ => None,
35    })
36}

38/// Returns the ident of the function argument, if it has one.
39pub fn fn_arg_ident(arg: &FnArg) -> Result<Ident, Error> {
40    if let FnArg::Typed(pat_type) = arg {
41        if let Pat::Ident(pat_ident) = *pat_type.pat.clone() {
42            return Ok(pat_ident.ident);
43        }
44    }
45    Err(Error::new(
46        arg.span(),
47        "argument in this form is not supported, use simple named arguments only",
48    ))
49}

51/// Returns a clone of the type from the FnArg.
52pub fn fn_arg_ref_type(arg: &FnArg, lifetime: Option<&Lifetime>) -> Result<Type, Error> {
53    if let FnArg::Typed(pat_type) = arg {
54        if !matches!(*pat_type.ty, Type::Reference(_)) {
55            Ok(Type::Reference(TypeReference {
56                and_token: And::default(),
57                lifetime: lifetime.cloned(),
58                mutability: None,
59                elem: pat_type.ty.clone(),
60            }))
61        } else {
62            Ok((*pat_type.ty).clone())
63        }
64    } else {
65        Err(Error::new(
66            arg.span(),
67            "argument in this form is not supported, use simple named arguments only",
68        ))
69    }
70}

72/// Returns a clone of FnArg with the type as a reference if the arg is a typed
73/// arg and its type is not already a reference.
74pub fn fn_arg_make_ref(arg: &FnArg, lifetime: Option<&Lifetime>) -> FnArg {
75    if let FnArg::Typed(pat_type) = arg {
76        if !matches!(*pat_type.ty, Type::Reference(_)) {
77            return FnArg::Typed(PatType {
78                attrs: pat_type.attrs.clone(),
79                pat: pat_type.pat.clone(),
80                colon_token: pat_type.colon_token,
81                ty: Box::new(Type::Reference(TypeReference {
82                    and_token: And::default(),
83                    lifetime: lifetime.cloned(),
84                    mutability: None,
85                    elem: pat_type.ty.clone(),
86                })),
87            });
88        }
89    }
90    arg.clone()
91}

93pub enum HasFnsItem {
94    Trait(ItemTrait),
95    Impl(ItemImpl),
96}
97
98impl HasFnsItem {
99    pub fn name(&'_ self) -> String {
100        match self {
101            HasFnsItem::Trait(t) => t.ident.to_string(),
102            HasFnsItem::Impl(i) => {
103                let ty = &i.self_ty;
104                quote!(#ty).to_string()
105            }
106        }
107    }
108
109    pub fn fns(&'_ self) -> Vec<Fn> {
110        match self {
111            HasFnsItem::Trait(t) => trait_methods(t)
112                .map(|m| Fn {
113                    ident: &m.sig.ident,
114                    attrs: &m.attrs,
115                    inputs: &m.sig.inputs,
116                    output: &m.sig.output,
117                })
118                .collect(),
119            HasFnsItem::Impl(i) => impl_pub_methods(i)
120                .map(|m| Fn {
121                    ident: &m.sig.ident,
122                    attrs: &m.attrs,
123                    inputs: &m.sig.inputs,
124                    output: &m.sig.output,
125                })
126                .collect(),
127        }
128    }
129}

131impl Parse for HasFnsItem {
132    fn parse(input: ParseStream) -> syn::Result<Self> {
133        _ = input.call(Attribute::parse_outer);
134        _ = input.parse::<Token![pub]>();
135        let lookahead = input.lookahead1();
136        if lookahead.peek(Token![trait]) {
137            let t = input.parse()?;
138            Ok(HasFnsItem::Trait(t))
139        } else if lookahead.peek(Token![impl]) {
140            let mut imp = input.parse()?;
141            flatten_associated_items_in_impl_fns(&mut imp);
142            Ok(HasFnsItem::Impl(imp))
143        } else {
144            Err(lookahead.error())
145        }
146    }
147}
148
149impl ToTokens for HasFnsItem {
150    fn to_tokens(&self, tokens: &mut TokenStream) {
151        match self {
152            HasFnsItem::Trait(t) => t.to_tokens(tokens),
153            HasFnsItem::Impl(i) => i.to_tokens(tokens),
154        }
155    }
156}

158pub struct Fn<'a> {
159    pub ident: &'a Ident,
160    pub attrs: &'a [Attribute],
161    pub inputs: &'a Punctuated<FnArg, Comma>,
162    pub output: &'a ReturnType,
163}
164
165impl<'a> Fn<'a> {
166    pub fn output(&self) -> Type {
167        let t = match self.output {
168            ReturnType::Default => quote!(()),
169            ReturnType::Type(_, typ) => match unpack_result(typ) {
170                Some((t, _)) => quote!(#t),
171                None => quote!(#typ),
172            },
173        };
174        Type::Verbatim(t)
175    }
176    pub fn try_output(&self, crate_path: &Path) -> Type {
177        let (t, e) = match self.output {
178            ReturnType::Default => (quote!(()), quote!(#crate_path::Error)),
179            ReturnType::Type(_, typ) => match unpack_result(typ) {
180                Some((t, e)) => (quote!(#t), quote!(#e)),
181                None => (quote!(#typ), quote!(#crate_path::Error)),
182            },
183        };
184        Type::Verbatim(quote! {
185            Result<
186                Result<#t, <#t as #crate_path::TryFromVal<#crate_path::Env, #crate_path::Val>>::Error>,
187                Result<#e, #crate_path::InvokeError>
188            >
189        })
190    }
191}
192
193fn unpack_result(typ: &Type) -> Option<(Type, Type)> {
194    match &typ {
195        Type::Path(TypePath { path, .. }) => {
196            if let Some(PathSegment {
197                ident,
198                arguments:
199                    PathArguments::AngleBracketed(AngleBracketedGenericArguments { args, .. }),
200            }) = path.segments.last()
201            {
202                let args = args.iter().collect::<Vec<_>>();
203                match (&ident.to_string()[..], args.as_slice()) {
204                    ("Result", [GenericArgument::Type(t), GenericArgument::Type(e)]) => {
205                        Some((t.clone(), e.clone()))
206                    }
207                    _ => None,
208                }
209            } else {
210                None
211            }
212        }
213        _ => None,
214    }
215}

217fn flatten_associated_items_in_impl_fns(imp: &mut ItemImpl) {
218    // TODO: Flatten associated consts used in functions.
219    // Flatten associated types used in functions.
220    let associated_types = imp
221        .items
222        .iter()
223        .filter_map(|item| match item {
224            ImplItem::Type(i) => Some((i.ident.clone(), i.ty.clone())),
225            _ => None,
226        })
227        .collect::<HashMap<_, _>>();
228    let fn_input_types = imp
229        .items
230        .iter_mut()
231        .filter_map(|item| match item {
232            ImplItem::Fn(f) => Some(f.sig.inputs.iter_mut().filter_map(|input| match input {
233                FnArg::Typed(t) => Some(&mut t.ty),
234                _ => None,
235            })),
236            _ => None,
237        })
238        .flatten();
239    for t in fn_input_types {
240        if let Type::Path(TypePath { qself: None, path }) = t.as_mut() {
241            let segments = &path.segments;
242            if segments.len() == 2
243                && segments.first() == Some(&PathSegment::from(format_ident!("Self")))
244            {
245                if let Some(PathSegment {
246                    arguments: PathArguments::None,
247                    ident,
248                }) = segments.get(1)
249                {
250                    if let Some(resolved_ty) = associated_types.get(ident) {
251                        *t.as_mut() = resolved_ty.clone();
252                    }
253                }
254            }
255        }
256    }
