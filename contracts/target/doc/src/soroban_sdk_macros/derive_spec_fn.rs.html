17#[allow(clippy::too_many_arguments)]
18pub fn derive_fn_spec(
19    ty: &Type,
20    ident: &Ident,
21    attrs: &[Attribute],
22    inputs: &Punctuated<FnArg, Comma>,
23    output: &ReturnType,
24    export: bool,
25) -> Result<TokenStream2, TokenStream2> {
26    // Collect errors as they are encountered and emit them at the end.
27    let mut errors = Vec::<Error>::new();
28
29    // Prepare the env input.
30    let env_input = inputs.first().and_then(|a| match a {
31        FnArg::Typed(pat_type) => {
32            let mut ty = &*pat_type.ty;
33            if let Type::Reference(TypeReference { elem, .. }) = ty {
34                ty = elem;
35            }
36            if let Type::Path(TypePath {
37                path: syn::Path { segments, .. },
38                ..
39            }) = ty
40            {
41                if segments.last().map_or(false, |s| s.ident == "Env") {
42                    Some(())
43                } else {
44                    None
45                }
46            } else {
47                None
48            }
49        }
50        FnArg::Receiver(_) => None,
51    });
52
53    // Prepare the argument inputs.
54    let spec_args: Vec<_> = inputs
55        .iter()
56        .skip(if env_input.is_some() { 1 } else { 0 })
57        .enumerate()
58        .map(|(i, a)| match a {
59            FnArg::Typed(pat_type) => {
60                let name = if let Pat::Ident(pat_ident) = *pat_type.pat.clone() {
61                    pat_ident.ident.to_string()
62                } else {
63                    errors.push(Error::new(a.span(), "argument not supported"));
64                    "".to_string()
65                };
66
67                // If fn is a __check_auth implementation, allow the first argument,
68                // signature_payload of type Bytes (32 size), to be a Hash.
69                let allow_hash = ident == "__check_auth" && i == 0;
70
71                match map_type(&pat_type.ty, allow_hash) {
72                    Ok(type_) => {
73                        let name = name.try_into().unwrap_or_else(|_| {
74                            const MAX: u32 = 30;
75                            errors.push(Error::new(
76                                a.span(),
77                                format!("argument name too long, max length {} characters", MAX),
78                            ));
79                            StringM::<MAX>::default()
80                        });
81                        ScSpecFunctionInputV0 {
82                            doc: "".try_into().unwrap(),
83                            name,
84                            type_,
85                        }
86                    }
87                    Err(e) => {
88                        errors.push(e);
89                        ScSpecFunctionInputV0 {
90                            doc: "".try_into().unwrap(),
91                            name: "arg".try_into().unwrap(),
92                            type_: ScSpecTypeDef::I32,
93                        }
94                    }
95                }
96            }
97            FnArg::Receiver(_) => {
98                errors.push(Error::new(a.span(), "self argument not supported"));
99                ScSpecFunctionInputV0 {
100                    doc: "".try_into().unwrap(),
101                    name: "".try_into().unwrap(),
102                    type_: ScSpecTypeDef::I32,
103                }
104            }
105        })
106        .collect();
107
108    // Prepare the output.
109    let spec_result = match output {
110        ReturnType::Type(_, ty) => vec![match map_type(ty, true) {
111            Ok(spec) => spec,
112            Err(e) => {
113                errors.push(e);
114                ScSpecTypeDef::I32
115            }
116        }],
117        ReturnType::Default => vec![],
118    };
119
120    // Generated code spec.
121    let name = &format!("{}", ident);
122    let spec_entry = ScSpecEntry::FunctionV0(ScSpecFunctionV0 {
123        doc: docs_from_attrs(attrs),
124        name: name.try_into().unwrap_or_else(|_| {
125            errors.push(Error::new(
126                ident.span(),
127                format!(
128                    "contract function name is too long: {}, max is {}",
129                    name.len(),
130                    SCSYMBOL_LIMIT,
131                ),
132            ));
133            ScSymbol::default()
134        }),
135        inputs: spec_args.try_into().unwrap_or_else(|_| {
136            const MAX: u32 = 10;
137            errors.push(Error::new(
138                inputs.iter().nth(MAX as usize).span(),
139                format!(
140                    "contract function has too many parameters, max count {} parameters",
141                    MAX,
142                ),
143            ));
144            VecM::<_, MAX>::default()
145        }),
146        outputs: spec_result.try_into().unwrap(),
147    });
148    let spec_xdr = spec_entry.to_xdr(DEFAULT_XDR_RW_LIMITS).unwrap();
149    let spec_xdr_lit = proc_macro2::Literal::byte_string(spec_xdr.as_slice());
150    let spec_xdr_len = spec_xdr.len();
151    let spec_ident = format_ident!("__SPEC_XDR_FN_{}", ident.to_string().to_uppercase());
152    let spec_fn_ident = format_ident!("spec_xdr_{}", ident.to_string());
153
154    // If errors have occurred, render them instead.
155    if !errors.is_empty() {
156        let compile_errors = errors.iter().map(Error::to_compile_error);
157        return Err(quote! { #(#compile_errors)* });
158    }
159
160    let export_attr = if export {
161        Some(quote! { #[cfg_attr(target_family = "wasm", link_section = "contractspecv0")] })
162    } else {
163        None
164    };
165
166    // Filter attributes to those that should be passed through to the generated code.
167    let attrs = attrs
168        .iter()
169        .filter(|attr| pass_through_attr_to_gen_code(attr))
170        .collect::<Vec<_>>();
171
172    // Generated code.
173    Ok(quote! {
174        #[doc(hidden)]
175        #[allow(non_snake_case)]
176        #[allow(non_upper_case_globals)]
177        #(#attrs)*
178        #export_attr
179        pub static #spec_ident: [u8; #spec_xdr_len] = #ty::#spec_fn_ident();
180
181        impl #ty {
182            #[allow(non_snake_case)]
183            #(#attrs)*
184            pub const fn #spec_fn_ident() -> [u8; #spec_xdr_len] {
185                *#spec_xdr_lit
186            }
187        }
188    })
189}
