11// These constants' values must match the definitions of the constants with the
12// same names in soroban_sdk::crypto::bls12_381.
13pub const FP_SERIALIZED_SIZE: u32 = 48;
14pub const FP2_SERIALIZED_SIZE: u32 = FP_SERIALIZED_SIZE * 2;
15pub const G1_SERIALIZED_SIZE: u32 = FP_SERIALIZED_SIZE * 2;
16pub const G2_SERIALIZED_SIZE: u32 = FP2_SERIALIZED_SIZE * 2;
17
18#[allow(clippy::too_many_lines)]
19pub fn map_type(t: &Type, allow_hash: bool) -> Result<ScSpecTypeDef, Error> {
20    match t {
21        Type::Path(TypePath {
22            qself: None,
23            path: Path { segments, .. },
24        }) => {
25            match segments.last() {
26                Some(PathSegment {
27                    ident,
28                    arguments: PathArguments::None,
29                }) => match &ident.to_string()[..] {
30                    "Val" => Ok(ScSpecTypeDef::Val),
31                    "u64" => Ok(ScSpecTypeDef::U64),
32                    "i64" => Ok(ScSpecTypeDef::I64),
33                    "u32" => Ok(ScSpecTypeDef::U32),
34                    "i32" => Ok(ScSpecTypeDef::I32),
35                    "u128" => Ok(ScSpecTypeDef::U128),
36                    "i128" => Ok(ScSpecTypeDef::I128),
37                    "U256" => Ok(ScSpecTypeDef::U256),
38                    "I256" => Ok(ScSpecTypeDef::I256),
39                    "bool" => Ok(ScSpecTypeDef::Bool),
40                    "Symbol" => Ok(ScSpecTypeDef::Symbol),
41                    "String" => Ok(ScSpecTypeDef::String),
42                    "Error" => Ok(ScSpecTypeDef::Error),
43                    "Bytes" => Ok(ScSpecTypeDef::Bytes),
44                    "Address" => Ok(ScSpecTypeDef::Address),
45                    "Timepoint" => Ok(ScSpecTypeDef::Timepoint),
46                    "Duration" => Ok(ScSpecTypeDef::Duration),
47                    // The BLS types defined below are represented in the contract's
48                    // interface by their underlying data types, i.e.
49                    // Fp/Fp2/G1Affine/G2Affine => BytesN<N>, Fr => U256. This approach
50                    // simplifies integration with contract development tooling, as it
51                    // avoids introducing new spec types for these BLS constructs.
52                    //
53                    // While this is functionally sound because the BLS types are
54                    // essentially newtypes over their inner representations, it means
55                    // that the specific semantic meaning of `G1Affine`, `G2Affine`, or
56                    // `Fr` is not directly visible in the compiled WASM interface. For
57                    // example, a contract function expecting a `G1Affine` will appear
58                    // in the WASM interface as expecting a `BytesN<96>`.
59                    //
60                    // Future enhancements might allow the macro to automatically deduce
61                    // and utilize the inner types for types defined using the New Type
62                    // Idiom. For more details, see the tracking issue for supporting
63                    // type aliases:
64                    // https://github.com/stellar/rs-soroban-sdk/issues/1063
65                    "Fp" => Ok(ScSpecTypeDef::BytesN(ScSpecTypeBytesN {
66                        n: FP_SERIALIZED_SIZE,
67                    })),
68                    "Fp2" => Ok(ScSpecTypeDef::BytesN(ScSpecTypeBytesN {
69                        n: FP2_SERIALIZED_SIZE,
70                    })),
71                    "G1Affine" => Ok(ScSpecTypeDef::BytesN(ScSpecTypeBytesN {
72                        n: G1_SERIALIZED_SIZE,
73                    })),
74                    "G2Affine" => Ok(ScSpecTypeDef::BytesN(ScSpecTypeBytesN {
75                        n: G2_SERIALIZED_SIZE,
76                    })),
77                    "Fr" => Ok(ScSpecTypeDef::U256),
78                    s => Ok(ScSpecTypeDef::Udt(ScSpecTypeUdt {
79                        name: s.try_into().map_err(|e| {
80                            Error::new(
81                                t.span(),
82                                format!("Udt name {:?} cannot be used in XDR spec: {}", s, e),
83                            )
84                        })?,
85                    })),
86                },
87                Some(PathSegment {
88                    ident,
89                    arguments: PathArguments::AngleBracketed(angle_bracketed),
90                }) => {
91                    let args = angle_bracketed.args.iter().collect::<Vec<_>>();
92                    match &ident.to_string()[..] {
93                        "Result" => {
94                            let (ok, err) = match args.as_slice() {
95                                [GenericArgument::Type(ok), GenericArgument::Type(err)] => (ok, err),
96                                [..] => Err(Error::new(
97                                    t.span(),
98                                    "incorrect number of generic arguments, expect two for Result<T, E>",
99                                ))?,
100                            };
101                            Ok(ScSpecTypeDef::Result(Box::new(ScSpecTypeResult {
102                                ok_type: Box::new(map_type(ok, false)?),
103                                error_type: Box::new(map_type(err, false)?),
104                            })))
105                        }
106                        "Option" => {
107                            let t = match args.as_slice() {
108                            [GenericArgument::Type(t)] => t,
109                            [..] => Err(Error::new(
110                                t.span(),
111                                "incorrect number of generic arguments, expect one for Option<T>",
112                            ))?,
113                        };
114                            Ok(ScSpecTypeDef::Option(Box::new(ScSpecTypeOption {
115                                value_type: Box::new(map_type(t, false)?),
116                            })))
117                        }
118                        "Vec" => {
119                            let t = match args.as_slice() {
120                                [GenericArgument::Type(t)] => t,
121                                [..] => Err(Error::new(
122                                    t.span(),
123                                    "incorrect number of generic arguments, expect one for Vec<T>",
124                                ))?,
125                            };
126                            Ok(ScSpecTypeDef::Vec(Box::new(ScSpecTypeVec {
127                                element_type: Box::new(map_type(t, false)?),
128                            })))
129                        }
130                        "Map" => {
131                            let (k, v) = match args.as_slice() {
132                                [GenericArgument::Type(k), GenericArgument::Type(v)] => (k, v),
133                                [..] => Err(Error::new(
134                                    t.span(),
135                                    "incorrect number of generic arguments, expect two for Map<K, V>",
136                                ))?,
137                            };
138                            Ok(ScSpecTypeDef::Map(Box::new(ScSpecTypeMap {
139                                key_type: Box::new(map_type(k, false)?),
140                                value_type: Box::new(map_type(v, false)?),
141                            })))
142                        }
143                        "BytesN" => {
144                            let n = match args.as_slice() {
145                                [GenericArgument::Const(Expr::Lit(ExprLit { lit: Lit::Int(int), .. }))] => int.base10_parse()?,
146                                [..] => Err(Error::new(
147                                    t.span(),
148                                    "incorrect number of generic arguments, expect one for BytesN<N>",
149                                ))?,
150                            };
151                            Ok(ScSpecTypeDef::BytesN(ScSpecTypeBytesN { n }))
152                        }
153                        "Hash" => {
154                            if allow_hash {
155                                let n = match args.as_slice() {
156                                    [GenericArgument::Const(Expr::Lit(ExprLit { lit: Lit::Int(int), .. }))] => int.base10_parse()?,
157                                    [..] => Err(Error::new(
158                                        t.span(),
159                                        "incorrect number of generic arguments, expect one for Hash<N>",
160                                    ))?,
161                                };
162                                Ok(ScSpecTypeDef::BytesN(ScSpecTypeBytesN { n }))
163                            } else {
164                                Err(Error::new(
165                                    t.span(),
166                                    "Hash<N> can only be used in contexts where there is a guarantee that the hash has been sourced from a secure cryptographic hash function",
167                                ))
168                            }
169                        }
170                        _ => Err(Error::new(
171                            angle_bracketed.span(),
172                            "generics unsupported on user-defined types in contract functions",
173                        ))?,
174                    }
175                }
176                _ => Err(Error::new(t.span(), "unsupported type"))?,
177            }
178        }
179        Type::Tuple(TypeTuple { elems, .. }) => {
180            let map_type_reject_hash =
181                |t: &Type| -> Result<ScSpecTypeDef, Error> { map_type(t, false) };
182            Ok(ScSpecTypeDef::Tuple(Box::new(ScSpecTypeTuple {
183                value_types: elems
184                    .iter()
185                    .map(map_type_reject_hash)
186                    .collect::<Result<Vec<ScSpecTypeDef>, Error>>()? // TODO: Implement conversion to VecM from iters to omit this collect.
187                    .try_into()
188                    .map_err(|e| {
189                        Error::new(
190                            t.span(),
191                            format!("tuple values cannot be used in XDR spec: {}", e),
192                        )
193                    })?,
194            })))
195        }
196        _ => Err(Error::new(t.span(), "unsupported type"))?,
197    }
198}
