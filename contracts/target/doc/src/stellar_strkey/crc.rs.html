75/// Checksum returns the 2-byte checksum for the provided data, in little endian
76/// byte-order.
77pub fn checksum(data: &[u8]) -> [u8; 2] {
78    let mut crc: u16 = 0;
79    for b in data.iter() {
80        crc = (crc << 8) ^ CRC16_TABLE[((crc >> 8) as u8 ^ *b) as usize];
81    }
82    [(crc & 0xff) as u8, (crc >> 8) as u8]
83}

91    #[test]
92    fn test_checksum() {
93        let cs = checksum(b"123456789");
94        assert_eq!(cs, [0xc3, 0x31]);
95        let cs = checksum(&[0x12, 0x34, 0x56, 0x78, 0x90]);
96        assert_eq!(cs, [0xe6, 0x48]);
97    }
