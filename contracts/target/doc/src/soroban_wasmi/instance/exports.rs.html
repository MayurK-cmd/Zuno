5/// An external item to a WebAssembly module.
6///
7/// This is returned from [`Instance::exports`](crate::Instance::exports)
8/// or [`Instance::get_export`](crate::Instance::get_export).
9#[derive(Debug, Copy, Clone)]
10pub enum Extern {
11    /// A WebAssembly global which acts like a [`Cell<T>`] of sorts, supporting `get` and `set` operations.
12    ///
13    /// [`Cell<T>`]: https://doc.rust-lang.org/core/cell/struct.Cell.html
14    Global(Global),
15    /// A WebAssembly table which is an array of funtion references.
16    Table(Table),
17    /// A WebAssembly linear memory.
18    Memory(Memory),
19    /// A WebAssembly function which can be called.
20    Func(Func),
21}
22
23impl From<Global> for Extern {
24    fn from(global: Global) -> Self {
25        Self::Global(global)
26    }
27}
28
29impl From<Table> for Extern {
30    fn from(table: Table) -> Self {
31        Self::Table(table)
32    }
33}
34
35impl From<Memory> for Extern {
36    fn from(memory: Memory) -> Self {
37        Self::Memory(memory)
38    }
39}
40
41impl From<Func> for Extern {
42    fn from(func: Func) -> Self {
43        Self::Func(func)
44    }
45}

47impl Extern {
48    /// Returns the underlying global variable if `self` is a global variable.
49    ///
50    /// Returns `None` otherwise.
51    pub fn into_global(self) -> Option<Global> {
52        if let Self::Global(global) = self {
53            return Some(global);
54        }
55        None
56    }
57
58    /// Returns the underlying table if `self` is a table.
59    ///
60    /// Returns `None` otherwise.
61    pub fn into_table(self) -> Option<Table> {
62        if let Self::Table(table) = self {
63            return Some(table);
64        }
65        None
66    }
67
68    /// Returns the underlying linear memory if `self` is a linear memory.
69    ///
70    /// Returns `None` otherwise.
71    pub fn into_memory(self) -> Option<Memory> {
72        if let Self::Memory(memory) = self {
73            return Some(memory);
74        }
75        None
76    }
77
78    /// Returns the underlying function if `self` is a function.
79    ///
80    /// Returns `None` otherwise.
81    pub fn into_func(self) -> Option<Func> {
82        if let Self::Func(func) = self {
83            return Some(func);
84        }
85        None
86    }
87
88    /// Returns the type associated with this [`Extern`].
89    ///
90    /// # Panics
91    ///
92    /// If this item does not belong to the `store` provided.
93    pub fn ty(&self, ctx: impl AsContext) -> ExternType {
94        match self {
95            Extern::Global(global) => global.ty(ctx).into(),
96            Extern::Table(table) => table.ty(ctx).into(),
97            Extern::Memory(memory) => memory.ty(ctx).into(),
98            Extern::Func(func) => func.ty(ctx).into(),
99        }
100    }
101}

176/// An exported WebAssembly value.
177///
178/// This type is primarily accessed from the [`Instance::exports`](crate::Instance::exports) method
179/// and describes what names and items are exported from a Wasm [`Instance`](crate::Instance).
180#[derive(Debug, Clone)]
181pub struct Export<'instance> {
182    /// The name of the exported item.
183    name: &'instance str,
184    /// The definition of the exported item.
185    definition: Extern,
186}
187
188impl<'instance> Export<'instance> {
189    /// Creates a new [`Export`] with the given `name` and `definition`.
190    pub(crate) fn new(name: &'instance str, definition: Extern) -> Export<'instance> {
191        Self { name, definition }
192    }
193
194    /// Returns the name by which this export is known.
195    pub fn name(&self) -> &'instance str {
196        self.name
197    }
198
199    /// Return the [`ExternType`] of this export.
200    ///
201    /// # Panics
202    ///
203    /// If `ctx` does not own this [`Export`].
204    pub fn ty(&self, ctx: impl AsContext) -> ExternType {
205        self.definition.ty(ctx)
206    }
207
208    /// Consume this [`Export`] and return the underlying [`Extern`].
209    pub fn into_extern(self) -> Extern {
210        self.definition
211    }
212
213    /// Returns the underlying [`Func`], if the [`Export`] is a function or `None` otherwise.
214    pub fn into_func(self) -> Option<Func> {
215        self.definition.into_func()
216    }
217
218    /// Returns the underlying [`Table`], if the [`Export`] is a table or `None` otherwise.
219    pub fn into_table(self) -> Option<Table> {
220        self.definition.into_table()
221    }
222
223    /// Returns the underlying [`Memory`], if the [`Export`] is a linear memory or `None` otherwise.
224    pub fn into_memory(self) -> Option<Memory> {
225        self.definition.into_memory()
226    }
227
228    /// Returns the underlying [`Global`], if the [`Export`] is a global variable or `None` otherwise.
229    pub fn into_global(self) -> Option<Global> {
230        self.definition.into_global()
231    }
232}

246    /// Prepares an item to match the expected iterator `Item` signature.
