1//! The `wasmi` virtual machine definitions.
2//!
3//! These closely mirror the WebAssembly specification definitions.
4//! The overall structure is heavily inspired by the `wasmtime` virtual
5//! machine architecture.
6//!
7//! # Example
8//!
9//! The following example shows a "Hello, World!"-like example of creating
10//! a Wasm module from some initial `.wat` contents, defining a simple host
11//! function and calling the exported Wasm function.
12//!
13//! The example was inspired by
14//! [Wasmtime's API example](https://docs.rs/wasmtime/0.39.1/wasmtime/).
15//!
16//! ```
17//! use soroban_wasmi as wasmi;
18//! use anyhow::{anyhow, Result};
19//! use wasmi::*;
20//!
21//! fn main() -> Result<()> {
22//!     // First step is to create the Wasm execution engine with some config.
23//!     // In this example we are using the default configuration.
24//!     let engine = Engine::default();
25//!     let wat = r#"
26//!         (module
27//!             (import "host" "hello" (func $host_hello (param i32)))
28//!             (func (export "hello")
29//!                 (call $host_hello (i32.const 3))
30//!             )
31//!         )
32//!     "#;
33//!     // Wasmi does not yet support parsing `.wat` so we have to convert
34//!     // out `.wat` into `.wasm` before we compile and validate it.
35//!     let wasm = wat::parse_str(&wat)?;
36//!     let module = Module::new(&engine, &mut &wasm[..])?;
37//!
38//!     // All Wasm objects operate within the context of a `Store`.
39//!     // Each `Store` has a type parameter to store host-specific data,
40//!     // which in this case we are using `42` for.
41//!     type HostState = u32;
42//!     let mut store = Store::new(&engine, 42);
43//!     let host_hello = Func::wrap(&mut store, |caller: Caller<'_, HostState>, param: i32| {
44//!         println!("Got {param} from WebAssembly");
45//!         println!("My host state is: {}", caller.data());
46//!     });
47//!
48//!     // In order to create Wasm module instances and link their imports
49//!     // and exports we require a `Linker`.
50//!     let mut linker = <Linker<HostState>>::new(&engine);
51//!     // Instantiation of a Wasm module requires defining its imports and then
52//!     // afterwards we can fetch exports by name, as well as asserting the
53//!     // type signature of the function with `get_typed_func`.
54//!     //
55//!     // Also before using an instance created this way we need to start it.
56//!     linker.define("host", "hello", host_hello)?;
57//!     let instance = linker
58//!         .instantiate(&mut store, &module)?
59//!         .start(&mut store)?;
60//!     let hello = instance.get_typed_func::<(), ()>(&store, "hello")?;
61//!
62//!     // And finally we can call the wasm!
63//!     hello.call(&mut store, ())?;
64//!
65//!     Ok(())
66//! }
67//! ```
