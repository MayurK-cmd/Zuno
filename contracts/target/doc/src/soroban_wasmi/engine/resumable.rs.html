14/// Returned by [`Engine`] methods for calling a function in a resumable way.
15///
16/// # Note
17///
18/// This is the base type for resumable call results and can be converted into
19/// either the dynamically typed [`ResumableCall`] or the statically typed
20/// [`TypedResumableCall`] that act as user facing API. Therefore this type
21/// must provide all the information necessary to be properly converted into
22/// either user facing types.
23#[derive(Debug)]
24pub(crate) enum ResumableCallBase<T> {
25    /// The resumable call has finished properly and returned a result.
26    Finished(T),
27    /// The resumable call encountered a host error and can be resumed.
28    Resumable(ResumableInvocation),
29}
30
31/// Returned by calling a [`Func`] in a resumable way.
32#[derive(Debug)]
33pub enum ResumableCall {
34    /// The resumable call has finished properly and returned a result.
35    Finished,
36    /// The resumable call encountered a host error and can be resumed.
37    Resumable(ResumableInvocation),
38}
39
40impl ResumableCall {
41    /// Creates a [`ResumableCall`] from the [`Engine`]'s base [`ResumableCallBase`].
42    pub(crate) fn new(call: ResumableCallBase<()>) -> Self {
43        match call {
44            ResumableCallBase::Finished(()) => Self::Finished,
45            ResumableCallBase::Resumable(invocation) => Self::Resumable(invocation),
46        }
47    }
48}

50/// State required to resume a [`Func`] invocation.
51#[derive(Debug)]
52pub struct ResumableInvocation {
53    /// The engine in use for the function invokation.
54    ///
55    /// # Note
56    ///
57    /// - This handle is required to resolve function types
58    ///   of both `func` and `host_func` fields as well as in
59    ///   the `Drop` impl to recycle the stack.
60    engine: Engine,
61    /// The underlying root function to be executed.
62    ///
63    /// # Note
64    ///
65    /// The results of this function must always match with the
66    /// results given when resuming the call.
67    func: Func,
68    /// The host function that returned a host error.
69    ///
70    /// # Note
71    ///
72    /// - This is required to receive its result values that are
73    ///   needed to be fed back in manually by the user. This way we
74    ///   avoid heap memory allocations.
75    /// - The results of this function must always match with the
76    ///   arguments given when resuming the call.
77    host_func: Func,
78    /// The host error that was returned by the `host_func` which
79    /// caused the resumable function invocation to break.
80    ///
81    /// # Note
82    ///
83    /// This might be useful to users of this API to inspect the
84    /// actual host error. This is therefore guaranteed to never
85    /// be a Wasm trap.
86    host_error: Trap,
87    /// The value and call stack in use by the [`ResumableInvocation`].
88    ///
89    /// # Note
90    ///
91    /// - We need to keep the stack around since the user might want to
92    ///   resume the execution.
93    /// - This stack is borrowed from the engine and needs to be given
94    ///   back to the engine when the [`ResumableInvocation`] goes out
95    ///   of scope.
96    pub(super) stack: Stack,
97}
98
99impl ResumableInvocation {
100    /// Creates a new [`ResumableInvocation`].
101    pub(super) fn new(
102        engine: Engine,
103        func: Func,
104        host_func: Func,
105        host_error: Trap,
106        stack: Stack,
107    ) -> Self {
108        Self {
109            engine,
110            func,
111            host_func,
112            host_error,
113            stack,
114        }
115    }
116
117    /// Replaces the internal stack with an empty one that has no heap allocations.
118    pub(super) fn take_stack(&mut self) -> Stack {
119        replace(&mut self.stack, Stack::empty())
120    }
121
122    /// Updates the [`ResumableInvocation`] with the new `host_func` and a `host_error`.
123    pub(super) fn update(&mut self, host_func: Func, host_error: Trap) {
124        self.host_func = host_func;
125        self.host_error = host_error;
126    }
127}
128
129impl Drop for ResumableInvocation {
130    fn drop(&mut self) {
131        let stack = self.take_stack();
132        self.engine.recycle_stack(stack);
133    }
134}

136impl ResumableInvocation {
137    /// Returns the host [`Func`] that returned the host error.
138    ///
139    /// # Note
140    ///
141    /// When using [`ResumableInvocation::resume`] the `inputs`
142    /// need to match the results of this host function so that
143    /// the function invocation can properly resume. For that
144    /// number and types of the values provided must match.
145    pub fn host_func(&self) -> Func {
146        self.host_func
147    }
148
149    /// Returns a shared reference to the encountered host error.
150    ///
151    /// # Note
152    ///
153    /// This is guaranteed to never be a Wasm trap.
154    pub fn host_error(&self) -> &Trap {
155        &self.host_error
156    }
157
158    /// Resumes the call to the [`Func`] with the given inputs.
159    ///
160    /// The result is written back into the `outputs` buffer upon success.
161    ///
162    /// Returns a resumable handle to the function invocation upon
163    /// enountering host errors with which it is possible to handle
164    /// the error and continue the execution as if no error occured.
165    ///
166    /// # Errors
167    ///
168    /// - If the function resumption returned a Wasm [`Trap`].
169    /// - If the types or the number of values in `inputs` does not match
170    ///   the types and number of result values of the errorneous host function.
171    /// - If the number of output values does not match the expected number of
172    ///   outputs required by the called function.
173    pub fn resume<T>(
174        self,
175        mut ctx: impl AsContextMut<UserState = T>,
176        inputs: &[Value],
177        outputs: &mut [Value],
178    ) -> Result<ResumableCall, Error> {
179        self.engine
180            .resolve_func_type(self.host_func().ty_dedup(ctx.as_context()), |func_type| {
181                func_type.match_results(inputs, true)
182            })?;
183        self.engine
184            .resolve_func_type(self.func.ty_dedup(ctx.as_context()), |func_type| {
185                func_type.match_results(outputs, false)?;
186                func_type.prepare_outputs(outputs);
187                <Result<(), Error>>::Ok(()) // TODO: why do we need types here?
188            })?;
189        self.engine
190            .clone()
191            .resume_func(ctx.as_context_mut(), self, inputs, outputs)
192            .map_err(Into::into)
193            .map(ResumableCall::new)
194    }
195}

197/// Returned by calling a [`TypedFunc`] in a resumable way.
198///
199/// [`TypedFunc`]: [`crate::TypedFunc`]
200#[derive(Debug)]
201pub enum TypedResumableCall<T> {
202    /// The resumable call has finished properly and returned a result.
203    Finished(T),
204    /// The resumable call encountered a host error and can be resumed.
205    Resumable(TypedResumableInvocation<T>),
206}
207
208impl<Results> TypedResumableCall<Results> {
209    /// Creates a [`TypedResumableCall`] from the [`Engine`]'s base [`ResumableCallBase`].
210    pub(crate) fn new(call: ResumableCallBase<Results>) -> Self {
211        match call {
212            ResumableCallBase::Finished(results) => Self::Finished(results),
213            ResumableCallBase::Resumable(invocation) => {
214                Self::Resumable(TypedResumableInvocation::new(invocation))
215            }
216        }
217    }
218}

220/// State required to resume a [`TypedFunc`] invocation.
221///
222/// [`TypedFunc`]: [`crate::TypedFunc`]
223pub struct TypedResumableInvocation<Results> {
224    invocation: ResumableInvocation,
225    /// The parameter and result typed encoded in Rust type system.
226    results: PhantomData<fn() -> Results>,
227}
228
229impl<Results> TypedResumableInvocation<Results> {
230    /// Creates a [`TypedResumableInvocation`] wrapper for the given [`ResumableInvocation`].
231    pub(crate) fn new(invocation: ResumableInvocation) -> Self {
232        Self {
233            invocation,
234            results: PhantomData,
235        }
236    }
237
238    /// Resumes the call to the [`TypedFunc`] with the given inputs.
239    ///
240    /// Returns a resumable handle to the function invocation upon
241    /// enountering host errors with which it is possible to handle
242    /// the error and continue the execution as if no error occured.
243    ///
244    /// # Errors
245    ///
246    /// - If the function resumption returned a Wasm [`Trap`].
247    /// - If the types or the number of values in `inputs` does not match
248    ///   the types and number of result values of the errorneous host function.
249    ///
250    /// [`TypedFunc`]: [`crate::TypedFunc`]
251    pub fn resume<T>(
252        self,
253        mut ctx: impl AsContextMut<UserState = T>,
254        inputs: &[Value],
255    ) -> Result<TypedResumableCall<Results>, Error>
256    where
257        Results: WasmResults,
258    {
259        self.engine
260            .resolve_func_type(self.host_func().ty_dedup(ctx.as_context()), |func_type| {
261                func_type.match_results(inputs, true)
262            })?;
263        self.engine
264            .clone()
265            .resume_func(
266                ctx.as_context_mut(),
267                self.invocation,
268                inputs,
269                <CallResultsTuple<Results>>::default(),
270            )
271            .map_err(Into::into)
272            .map(TypedResumableCall::new)
273    }
274}

279    fn deref(&self) -> &Self::Target {
280        &self.invocation
281    }
