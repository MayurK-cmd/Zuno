28/// The internal `wasmi` bytecode that is stored for Wasm functions.
29///
30/// # Note
31///
32/// This representation slightly differs from WebAssembly instructions.
33///
34/// For example the `BrTable` instruction is unrolled into separate instructions
35/// each representing either the `BrTable` head or one of its branching targets.
36#[derive(Debug, Copy, Clone, PartialEq, Eq)]
37pub enum Instruction {
38    LocalGet(LocalDepth),
39    LocalSet(LocalDepth),
40    LocalTee(LocalDepth),
41    /// An unconditional branch.
42    Br(BranchOffset),
43    /// Branches if the top-most stack value is equal to zero.
44    BrIfEqz(BranchOffset),
45    /// Branches if the top-most stack value is _not_ equal to zero.
46    BrIfNez(BranchOffset),
47    /// An unconditional branch.
48    ///
49    /// This operation also adjust the underlying value stack if necessary.
50    ///
51    /// # Encoding
52    ///
53    /// This [`Instruction`] must be followed by a [`Instruction::Return`]
54    /// which stores information about the [`DropKeep`] behavior of the
55    /// [`Instruction::Br`]. The [`Instruction::Return`] will never be executed
56    /// and only acts as parameter storage for this instruction.
57    BrAdjust(BranchOffset),
58    /// Branches if the top-most stack value is _not_ equal to zero.
59    ///
60    /// This operation also adjust the underlying value stack if necessary.
61    ///
62    /// # Encoding
63    ///
64    /// This [`Instruction`] must be followed by a [`Instruction::Return`]
65    /// which stores information about the [`DropKeep`] behavior of the
66    /// [`Instruction::BrIfNez`]. The [`Instruction::Return`] will never be executed
67    /// and only acts as parameter storage for this instruction.
68    BrAdjustIfNez(BranchOffset),
69    /// Branch table with a set number of branching targets.
70    ///
71    /// # Encoding
72    ///
73    /// This [`Instruction`] must be followed by exactly as many unconditional
74    /// branch instructions as determined by [`BranchTableTargets`]. Branch
75    /// instructions that may follow are [`Instruction::Br] and [`Instruction::Return`].
76    BrTable(BranchTableTargets),
77    Unreachable,
78    ConsumeFuel(BlockFuel),
79    Return(DropKeep),
80    ReturnIfNez(DropKeep),
81    /// Tail calls an internal (compiled) function.
82    ///
83    /// # Note
84    ///
85    /// This instruction can be used for calls to functions that are engine internal
86    /// (or compiled) and acts as an optimization for those common cases.
87    ///
88    /// # Encoding
89    ///
90    /// This [`Instruction`] must be followed by an [`Instruction::Return`] that
91    /// encodes the [`DropKeep`] parameter. Note that the [`Instruction::Return`]
92    /// only acts as a storage for the parameter of the [`Instruction::ReturnCall`]
93    /// and will never be executed by itself.
94    ReturnCallInternal(CompiledFunc),
95    /// Tail calling `func`.
96    ///
97    /// # Note
98    ///
99    /// Since [`Instruction::ReturnCallInternal`] should be used for all functions internal
100    /// (or compiled) to the engine this instruction should mainly be used for tail calling
101    /// imported functions. However, it is a general form that can technically be used
102    /// for both.
103    ///
104    /// # Encoding
105    ///
106    /// This [`Instruction`] must be followed by an [`Instruction::Return`] that
107    /// encodes the [`DropKeep`] parameter. Note that the [`Instruction::Return`]
108    /// only acts as a storage for the parameter of the [`Instruction::ReturnCall`]
109    /// and will never be executed by itself.
110    ReturnCall(FuncIdx),
111    /// Tail calling a function indirectly.
112    ///
113    /// # Encoding
114    ///
115    /// This [`Instruction`] must be followed by an [`Instruction::Return`] that
116    /// encodes the [`DropKeep`] parameter as well as an [`Instruction::TableGet`]
117    /// that encodes the [`TableIdx`] parameter. Note that both, [`Instruction::Return`]
118    /// and [`Instruction::TableGet`] only act as a storage for parameters to the
119    /// [`Instruction::ReturnCallIndirect`] and will never be executed by themselves.
120    ReturnCallIndirect(SignatureIdx),
121    /// Calls an internal (compiled) function.
122    ///
123    /// # Note
124    ///
125    /// This instruction can be used for calls to functions that are engine internal
126    /// (or compiled) and acts as an optimization for those common cases.
127    CallInternal(CompiledFunc),
128    /// Calls the function.
129    ///
130    /// # Note
131    ///
132    /// Since [`Instruction::CallInternal`] should be used for all functions internal
133    /// (or compiled) to the engine this instruction should mainly be used for calling
134    /// imported functions. However, it is a general form that can technically be used
135    /// for both.
136    Call(FuncIdx),
137    /// Calling a function indirectly.
138    ///
139    /// # Encoding
140    ///
141    /// This [`Instruction`] must be followed by an [`Instruction::TableGet`]
142    /// that encodes the [`TableIdx`] parameter. Note that the [`Instruction::TableGet`]
143    /// only acts as a storage for the parameter of the [`Instruction::CallIndirect`]
144    /// and will never be executed by itself.
145    CallIndirect(SignatureIdx),
146    Drop,
147    Select,
148    GlobalGet(GlobalIdx),
149    GlobalSet(GlobalIdx),
150    I32Load(AddressOffset),
151    I64Load(AddressOffset),
152    F32Load(AddressOffset),
153    F64Load(AddressOffset),
154    I32Load8S(AddressOffset),
155    I32Load8U(AddressOffset),
156    I32Load16S(AddressOffset),
157    I32Load16U(AddressOffset),
158    I64Load8S(AddressOffset),
159    I64Load8U(AddressOffset),
160    I64Load16S(AddressOffset),
161    I64Load16U(AddressOffset),
162    I64Load32S(AddressOffset),
163    I64Load32U(AddressOffset),
164    I32Store(AddressOffset),
165    I64Store(AddressOffset),
166    F32Store(AddressOffset),
167    F64Store(AddressOffset),
168    I32Store8(AddressOffset),
169    I32Store16(AddressOffset),
170    I64Store8(AddressOffset),
171    I64Store16(AddressOffset),
172    I64Store32(AddressOffset),
173    MemorySize,
174    MemoryGrow,
175    MemoryFill,
176    MemoryCopy,
177    MemoryInit(DataSegmentIdx),
178    DataDrop(DataSegmentIdx),
179    TableSize(TableIdx),
180    TableGrow(TableIdx),
181    TableFill(TableIdx),
182    TableGet(TableIdx),
183    TableSet(TableIdx),
184    /// Copies elements from one table to another.
185    ///
186    /// # Note
187    ///
188    /// It is also possible to copy elements within the same table.
189    ///
190    /// # Encoding
191    ///
192    /// The [`TableIdx`] referred to by the [`Instruction::TableCopy`]
193    /// represents the `dst` (destination) table. The [`Instruction::TableCopy`]
194    /// must be followed by an [`Instruction::TableGet`] which stores a
195    /// [`TableIdx`] that refers to the `src` (source) table.
196    TableCopy(TableIdx),
197    /// Initializes a table given an [`ElementSegmentIdx`].
198    ///
199    /// # Encoding
200    ///
201    /// The [`Instruction::TableInit`] must be followed by an
202    /// [`Instruction::TableGet`] which stores a [`TableIdx`]
203    /// that refers to the table to be initialized.
204    TableInit(ElementSegmentIdx),
205    ElemDrop(ElementSegmentIdx),
206    RefFunc(FuncIdx),
207    /// A 32-bit constant value.
208    Const32([u8; 4]),
209    /// A 64-bit integer value losslessly encoded as 32-bit integer.
210    ///
211    /// Upon execution the 32-bit integer is sign-extended to the 64-bit integer.
212    ///
213    /// # Note
214    ///
215    /// This is a space-optimized variant of [`Instruction::ConstRef`] but can
216    /// only used for small integer values that fit into a 32-bit integer value.
217    I64Const32(i32),
218    /// A 64-bit float value losslessly encoded as 32-bit float.
219    ///
220    /// Upon execution the 32-bit float is promoted to the 64-bit float.
221    ///
222    /// # Note
223    ///
224    /// This is a space-optimized variant of [`Instruction::ConstRef`] but can
225    /// only used for certain float values that fit into a 32-bit float value.
226    F64Const32(F64Const32),
227    /// Pushes a constant value onto the stack.
228    ///
229    /// The constant value is referred to indirectly by the [`ConstRef`].
230    ConstRef(ConstRef),
231    I32Eqz,
232    I32Eq,

365impl Instruction {
366    /// Creates an [`Instruction::Const32`] from the given `i32` constant value.
367    pub fn i32_const(value: i32) -> Self {
368        Self::Const32(value.to_ne_bytes())
369    }
370
371    /// Creates an [`Instruction::Const32`] from the given `f32` constant value.
372    pub fn f32_const(value: F32) -> Self {
373        Self::Const32(value.to_bits().to_ne_bytes())
374    }
375
376    /// Creates a new `local.get` instruction from the given local depth.
377    ///
378    /// # Errors
379    ///
380    /// If the `local_depth` is out of bounds as local depth index.
381    pub fn local_get(local_depth: u32) -> Result<Self, TranslationError> {
382        Ok(Self::LocalGet(LocalDepth::from(local_depth)))
383    }
384
385    /// Creates a new `local.set` instruction from the given local depth.
386    ///
387    /// # Errors
388    ///
389    /// If the `local_depth` is out of bounds as local depth index.
390    pub fn local_set(local_depth: u32) -> Result<Self, TranslationError> {
391        Ok(Self::LocalSet(LocalDepth::from(local_depth)))
392    }
393
394    /// Creates a new `local.tee` instruction from the given local depth.
395    ///
396    /// # Errors
397    ///
398    /// If the `local_depth` is out of bounds as local depth index.
399    pub fn local_tee(local_depth: u32) -> Result<Self, TranslationError> {
400        Ok(Self::LocalTee(LocalDepth::from(local_depth)))
401    }
402
403    /// Convenience method to create a new `ConsumeFuel` instruction.
404    pub fn consume_fuel(amount: u64) -> Result<Self, TranslationError> {
405        let block_fuel = BlockFuel::try_from(amount)?;
406        Ok(Self::ConsumeFuel(block_fuel))
407    }
408
409    /// Increases the fuel consumption of the [`ConsumeFuel`] instruction by `delta`.
410    ///
411    /// # Panics
412    ///
413    /// - If `self` is not a [`ConsumeFuel`] instruction.
414    /// - If the new fuel consumption overflows the internal `u64` value.
415    ///
416    /// [`ConsumeFuel`]: Instruction::ConsumeFuel
417    pub fn bump_fuel_consumption(&mut self, delta: u64) -> Result<(), TranslationError> {
418        match self {
419            Self::ConsumeFuel(block_fuel) => block_fuel.bump_by(delta),
420            instr => panic!("expected Instruction::ConsumeFuel but found: {instr:?}"),
421        }
422    }
