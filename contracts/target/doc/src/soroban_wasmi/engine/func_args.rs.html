1//! API using the Rust type system to guide host function trampoline execution.
2
3use crate::{value::WithType, Value};
4use core::cmp;
5use wasmi_core::{DecodeUntypedSlice, EncodeUntypedSlice, UntypedError, UntypedValue};
6
7/// Used to decode host function parameters.
8#[derive(Debug)]
9pub struct FuncParams<'a> {
10    /// Slice holding the raw (encoded but untyped) parameters
11    /// of the host function invocation before the call and the
12    /// results of the host function invocation after the call.
13    ///
14    /// Therefore the length of the slice must be large enough
15    /// to hold all parameters and all results but not both at
16    /// the same time.
17    params_results: &'a mut [UntypedValue],
18    /// The length of the expected parameters of the function invocation.
19    len_params: usize,
20    /// The length of the expected results of the function invocation.
21    len_results: usize,
22}
23
24/// Used to encode host function results.
25#[derive(Debug)]
26pub struct FuncResults<'a> {
27    results: &'a mut [UntypedValue],
28}
29
30impl<'a> FuncResults<'a> {
31    /// Create new [`FuncResults`] from the given `results` slice.
32    fn new(results: &'a mut [UntypedValue]) -> Self {
33        Self { results }
34    }
35
36    /// Encodes the results of the host function invocation as `T`.
37    ///
38    /// # Panics
39    ///
40    /// If the number of results dictated by `T` does not match the expected amount.
41    pub fn encode_results<T>(self, values: T) -> FuncFinished
42    where
43        T: EncodeUntypedSlice,
44    {
45        UntypedValue::encode_slice::<T>(self.results, values)
46            .unwrap_or_else(|error| panic!("encountered unexpected invalid tuple length: {error}"));
47        FuncFinished {}
48    }
49
50    /// Encodes the results of the host function invocation given the `values` slice.
51    ///
52    /// # Panics
53    ///
54    /// If the number of expected results does not match the length of `values`.
55    pub fn encode_results_from_slice(self, values: &[Value]) -> Result<FuncFinished, UntypedError> {
56        assert_eq!(self.results.len(), values.len());
57        self.results.iter_mut().zip(values).for_each(|(dst, src)| {
58            *dst = src.clone().into();
59        });
60        Ok(FuncFinished {})
61    }
62}
63
64/// Used to guarantee by the type system that this API has been used correctly.
65///
66/// Ensures at compile time that host functions always call
67/// [`FuncParams::decode_params`] or [`FuncParams::decode_params_into_slice`]
68/// followed by
69/// [`FuncResults::encode_results`] or [`FuncResults::encode_results_from_slice`]
70/// at the end of their execution.
71#[derive(Debug)]
72pub struct FuncFinished {}
73
74impl<'a> FuncParams<'a> {
75    /// Create new [`FuncParams`].
76    ///
77    /// # Panics
78    ///
79    /// If the length of hte `params_results` slice does not match the maximum
80    /// of the `len_params` and `Len_results`.
81    pub(super) fn new(
82        params_results: &'a mut [UntypedValue],
83        len_params: usize,
84        len_results: usize,
85    ) -> Self {
86        assert_eq!(params_results.len(), cmp::max(len_params, len_results));
87        Self {
88            params_results,
89            len_params,
90            len_results,
91        }
92    }
93
94    /// Returns a slice over the untyped function parameters.
95    fn params(&self) -> &[UntypedValue] {
96        &self.params_results[..self.len_params]
97    }
98
99    /// Decodes and returns the executed host function parameters as `T`.
100    ///
101    /// # Panics
102    ///
103    /// If the number of function parameters dictated by `T` does not match.
104    pub fn decode_params<T>(self) -> (T, FuncResults<'a>)
105    where
106        T: DecodeUntypedSlice,
107    {
108        let decoded = UntypedValue::decode_slice::<T>(self.params())
109            .unwrap_or_else(|error| panic!("encountered unexpected invalid tuple length: {error}"));
110        let results = self.into_func_results();
111        (decoded, results)
112    }
113
114    /// Decodes and stores the executed host functions parameters into `values`.
115    ///
116    /// # Panics
117    ///
118    /// If the number of host function parameters and items in `values` does not match.
119    pub fn decode_params_into_slice(
120        self,
121        values: &mut [Value],
122    ) -> Result<FuncResults<'a>, UntypedError> {
123        assert_eq!(self.params().len(), values.len());
124        self.params().iter().zip(values).for_each(|(src, dst)| {
125            *dst = src.with_type(dst.ty());
126        });
127        let results = self.into_func_results();
128        Ok(results)
129    }
130
131    /// Consumes `self` to return the [`FuncResults`] out of it.
132    fn into_func_results(self) -> FuncResults<'a> {
133        FuncResults::new(&mut self.params_results[..self.len_results])
134    }
135}
