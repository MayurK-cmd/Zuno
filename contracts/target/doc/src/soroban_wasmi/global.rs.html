7/// A raw index to a global variable entity.
8#[derive(Debug, Copy, Clone, PartialEq, Eq, PartialOrd, Ord)]
9pub struct GlobalIdx(u32);
10
11impl ArenaIndex for GlobalIdx {
12    fn into_usize(self) -> usize {
13        self.0 as usize
14    }
15
16    fn from_usize(value: usize) -> Self {
17        let value = value.try_into().unwrap_or_else(|error| {
18            panic!("index {value} is out of bounds as global index: {error}")
19        });
20        Self(value)
21    }
22}

24/// An error that may occur upon operating on global variables.
25#[derive(Debug)]
26#[non_exhaustive]
27pub enum GlobalError {
28    /// Occurs when trying to write to an immutable global variable.
29    ImmutableWrite,
30    /// Occurs when trying writing a value with mismatching type to a global variable.
31    TypeMismatch {
32        /// The type of the global variable.
33        expected: ValueType,
34        /// The type of the new value that mismatches the type of the global variable.
35        encountered: ValueType,
36    },
37    /// Occurs when a global type does not satisfy the constraints of another.
38    UnsatisfyingGlobalType {
39        /// The unsatisfying [`GlobalType`].
40        unsatisfying: GlobalType,
41        /// The required [`GlobalType`].
42        required: GlobalType,
43    },
44}
45
46impl Display for GlobalError {
47    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
48        match self {
49            Self::ImmutableWrite => write!(f, "tried to write to immutable global variable"),
50            Self::TypeMismatch {
51                expected,
52                encountered,
53            } => {
54                write!(
55                    f,
56                    "type mismatch upon writing global variable. \
57                    expected {expected:?} but encountered {encountered:?}.",
58                )
59            }
60            Self::UnsatisfyingGlobalType {
61                unsatisfying,
62                required,
63            } => {
64                write!(
65                    f,
66                    "global type {unsatisfying:?} does not \
67                    satisfy requirements of {required:?}",
68                )
69            }
70        }
71    }
72}

74/// The mutability of a global variable.
75#[derive(Debug, Copy, Clone, PartialEq, Eq)]
76pub enum Mutability {
77    /// The value of the global variable is a constant.
78    Const,
79    /// The value of the global variable is mutable.
80    Var,
81}
82
83impl Mutability {
84    /// Returns `true` if this mutability is [`Mutability::Const`].
85    pub fn is_const(&self) -> bool {
86        matches!(self, Self::Const)
87    }
88
89    /// Returns `true` if this mutability is [`Mutability::Var`].
90    pub fn is_mut(&self) -> bool {
91        matches!(self, Self::Var)
92    }
93}

95/// The type of a global variable.
96#[derive(Debug, Copy, Clone, PartialEq, Eq)]
97pub struct GlobalType {
98    /// The value type of the global variable.
99    content: ValueType,
100    /// The mutability of the global variable.
101    mutability: Mutability,
102}
103
104impl GlobalType {
105    /// Creates a new [`GlobalType`] from the given [`ValueType`] and [`Mutability`].
106    pub fn new(content: ValueType, mutability: Mutability) -> Self {
107        Self {
108            content,
109            mutability,
110        }
111    }
112
113    /// Returns the [`ValueType`] of the global variable.
114    pub fn content(&self) -> ValueType {
115        self.content
116    }
117
118    /// Returns the [`Mutability`] of the global variable.
119    pub fn mutability(&self) -> Mutability {
120        self.mutability
121    }
122
123    /// Checks if `self` satisfies the given `GlobalType`.
124    ///
125    /// # Errors
126    ///
127    /// - If the initial limits of the `required` [`GlobalType`] are greater than `self`.
128    /// - If the maximum limits of the `required` [`GlobalType`] are greater than `self`.
129    pub(crate) fn satisfies(&self, required: &GlobalType) -> Result<(), GlobalError> {
130        if self != required {
131            return Err(GlobalError::UnsatisfyingGlobalType {
132                unsatisfying: *self,
133                required: *required,
134            });
135        }
136        Ok(())
137    }
138}

140/// A global variable entity.
141#[derive(Debug)]
142pub struct GlobalEntity {
143    /// The current value of the global variable.
144    value: UntypedValue,
145    /// The type of the global variable.
146    ty: GlobalType,
147}
148
149impl GlobalEntity {
150    /// Creates a new global entity with the given initial value and mutability.
151    pub fn new(initial_value: Value, mutability: Mutability) -> Self {
152        Self {
153            ty: GlobalType::new(initial_value.ty(), mutability),
154            value: initial_value.into(),
155        }
156    }
157
158    /// Returns the [`GlobalType`] of the global variable.
159    pub fn ty(&self) -> GlobalType {
160        self.ty
161    }
162
163    /// Sets a new value to the global variable.
164    ///
165    /// # Errors
166    ///
167    /// - If the global variable is immutable.
168    /// - If there is a type mismatch between the global variable and the new value.
169    pub fn set(&mut self, new_value: Value) -> Result<(), GlobalError> {
170        if !self.ty().mutability().is_mut() {
171            return Err(GlobalError::ImmutableWrite);
172        }
173        if self.ty().content() != new_value.ty() {
174            return Err(GlobalError::TypeMismatch {
175                expected: self.ty().content(),
176                encountered: new_value.ty(),
177            });
178        }
179        self.set_untyped(new_value.into());
180        Ok(())
181    }
182
183    /// Sets a new untyped value for the global variable.
184    ///
185    /// # Note
186    ///
187    /// This is an inherently unsafe API and only exists to allow
188    /// for efficient `global.set` through the interpreter which is
189    /// safe since the interpreter only handles validated Wasm code
190    /// where the checks in [`Global::set`] cannot fail.
191    pub(crate) fn set_untyped(&mut self, new_value: UntypedValue) {
192        self.value = new_value;
193    }
194
195    /// Returns the current value of the global variable.
196    pub fn get(&self) -> Value {
197        self.get_untyped().with_type(self.ty().content())
198    }
199
200    /// Returns the current untyped value of the global variable.
201    pub(crate) fn get_untyped(&self) -> UntypedValue {
202        self.value
203    }
204
205    /// Returns a pointer to the untyped value of the global variable.
206    pub(crate) fn get_untyped_ptr(&mut self) -> NonNull<UntypedValue> {
207        NonNull::from(&mut self.value)
208    }
209}

211/// A Wasm global variable reference.
212#[derive(Debug, Copy, Clone)]
213#[repr(transparent)]
214pub struct Global(Stored<GlobalIdx>);
215
216impl Global {
217    /// Creates a new stored global variable reference.
218    ///
219    /// # Note
220    ///
221    /// This API is primarily used by the [`Store`] itself.
222    ///
223    /// [`Store`]: [`crate::Store`]
224    pub(super) fn from_inner(stored: Stored<GlobalIdx>) -> Self {
225        Self(stored)
226    }
227
228    /// Returns the underlying stored representation.
229    pub(super) fn as_inner(&self) -> &Stored<GlobalIdx> {
230        &self.0
231    }
232
233    /// Creates a new global variable to the store.
234    pub fn new(mut ctx: impl AsContextMut, initial_value: Value, mutability: Mutability) -> Self {
235        ctx.as_context_mut()
236            .store
237            .inner
238            .alloc_global(GlobalEntity::new(initial_value, mutability))
239    }
240
241    /// Returns the [`GlobalType`] of the global variable.
242    pub fn ty(&self, ctx: impl AsContext) -> GlobalType {
243        ctx.as_context().store.inner.resolve_global(self).ty()
244    }
245
246    /// Sets a new value to the global variable.
247    ///
248    /// # Errors
249    ///
250    /// - If the global variable is immutable.
251    /// - If there is a type mismatch between the global variable and the new value.
252    ///
253    /// # Panics
254    ///
255    /// Panics if `ctx` does not own this [`Global`].
256    pub fn set(&self, mut ctx: impl AsContextMut, new_value: Value) -> Result<(), GlobalError> {
257        ctx.as_context_mut()
258            .store
259            .inner
260            .resolve_global_mut(self)
261            .set(new_value)
262    }
263
264    /// Returns the current value of the global variable.
265    ///
266    /// # Panics
267    ///
268    /// Panics if `ctx` does not own this [`Global`].
269    pub fn get(&self, ctx: impl AsContext) -> Value {
270        ctx.as_context().store.inner.resolve_global(self).get()
271    }
272}
