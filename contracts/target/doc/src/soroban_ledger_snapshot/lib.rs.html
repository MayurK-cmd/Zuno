23/// Ledger snapshot stores a snapshot of a ledger that can be restored for use
24/// in environments as a [`LedgerInfo`] and a [`SnapshotSource`].
25#[serde_as]
26#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
27#[serde(rename_all = "snake_case")]
28pub struct LedgerSnapshot {
29    pub protocol_version: u32,
30    pub sequence_number: u32,
31    pub timestamp: u64,
32    #[serde_as(as = "serde_with::hex::Hex")]
33    pub network_id: [u8; 32],
34    pub base_reserve: u32,
35    pub min_persistent_entry_ttl: u32,
36    pub min_temp_entry_ttl: u32,
37    pub max_entry_ttl: u32,
38    pub ledger_entries: Vec<(Box<LedgerKey>, (Box<LedgerEntry>, Option<u32>))>,
39}
40
41impl LedgerSnapshot {
42    // Create a ledger snapshot from ledger info and a set of entries.
43    pub fn from<'a>(
44        info: LedgerInfo,
45        entries: impl IntoIterator<Item = (&'a Box<LedgerKey>, (&'a Box<LedgerEntry>, Option<u32>))>,
46    ) -> Self {
47        let mut s = Self::default();
48        s.set_ledger_info(info);
49        s.set_entries(entries);
50        s
51    }
52
53    /// Update the snapshot with the state within the given [`Host`].
54    ///
55    /// The ledger info of the host will overwrite the ledger info in the
56    /// snapshot.  The entries in the host's storage will overwrite entries in
57    /// the snapshot. Existing entries in the snapshot that are untouched by the
58    /// host will remain.
59    pub fn update(&mut self, host: &Host) {
60        let _result = host.with_ledger_info(|li| {
61            self.set_ledger_info(li.clone());
62            Ok(())
63        });
64        let _result = host.with_mut_storage(|s| {
65            self.update_entries(&s.map);
66            Ok(())
67        });
68    }
69
70    // Get the ledger info in the snapshot.
71    pub fn ledger_info(&self) -> LedgerInfo {
72        LedgerInfo {
73            protocol_version: self.protocol_version,
74            sequence_number: self.sequence_number,
75            timestamp: self.timestamp,
76            network_id: self.network_id,
77            base_reserve: self.base_reserve,
78            min_persistent_entry_ttl: self.min_persistent_entry_ttl,
79            min_temp_entry_ttl: self.min_temp_entry_ttl,
80            max_entry_ttl: self.max_entry_ttl,
81        }
82    }
83
84    /// Set the ledger info in the snapshot.
85    pub fn set_ledger_info(&mut self, info: LedgerInfo) {
86        self.protocol_version = info.protocol_version;
87        self.sequence_number = info.sequence_number;
88        self.timestamp = info.timestamp;
89        self.network_id = info.network_id;
90        self.base_reserve = info.base_reserve;
91        self.min_persistent_entry_ttl = info.min_persistent_entry_ttl;
92        self.min_temp_entry_ttl = info.min_temp_entry_ttl;
93        self.max_entry_ttl = info.max_entry_ttl;
94    }
95
96    /// Get the entries in the snapshot.
97    pub fn entries(
98        &self,
99    ) -> impl IntoIterator<Item = (&Box<LedgerKey>, &(Box<LedgerEntry>, Option<u32>))> {
100        self.ledger_entries.iter().map(|(k, v)| (k, v))
101    }
102
103    /// Replace the entries in the snapshot with the entries in the iterator.
104    pub fn set_entries<'a>(
105        &mut self,
106        entries: impl IntoIterator<Item = (&'a Box<LedgerKey>, (&'a Box<LedgerEntry>, Option<u32>))>,
107    ) {
108        self.ledger_entries.clear();
109        for (k, e) in entries {
110            self.ledger_entries.push((k.clone(), (e.0.clone(), e.1)));
111        }
112    }
113
114    /// Update entries in the snapshot by adding or replacing any entries that
115    /// have entry in the input iterator, or removing any that does not have an
116    /// entry.
117    pub fn update_entries<'a>(
118        &mut self,
119        entries: impl IntoIterator<Item = &'a (Rc<LedgerKey>, Option<(Rc<LedgerEntry>, Option<u32>)>)>,
120    ) {
121        for (k, e) in entries {
122            let i = self.ledger_entries.iter().position(|(ik, _)| **ik == **k);
123            if let Some((entry, live_until_ledger)) = e {
124                let new = (
125                    Box::new((**k).clone()),
126                    (Box::new((**entry).clone()), *live_until_ledger),
127                );
128                if let Some(i) = i {
129                    self.ledger_entries[i] = new;
130                } else {
131                    self.ledger_entries.push(new);
132                }
133            } else if let Some(i) = i {
134                self.ledger_entries.swap_remove(i);
135            }
136        }
137    }
138}

140impl LedgerSnapshot {
141    // Read in a [`LedgerSnapshot`] from a reader.
142    pub fn read(r: impl Read) -> Result<LedgerSnapshot, Error> {
143        Ok(serde_json::from_reader::<_, LedgerSnapshot>(r)?)
144    }
145
146    // Read in a [`LedgerSnapshot`] from a file.
147    pub fn read_file(p: impl AsRef<Path>) -> Result<LedgerSnapshot, Error> {
148        Self::read(File::open(p)?)
149    }
150
151    // Write a [`LedgerSnapshot`] to a writer.
152    pub fn write(&self, w: impl Write) -> Result<(), Error> {
153        Ok(serde_json::to_writer_pretty(w, self)?)
154    }
155
156    // Write a [`LedgerSnapshot`] to file.
157    pub fn write_file(&self, p: impl AsRef<Path>) -> Result<(), Error> {
158        let p = p.as_ref();
159        if let Some(dir) = p.parent() {
160            if !dir.exists() {
161                create_dir_all(dir)?;
162            }
163        }
164        self.write(File::create(p)?)
165    }
166}

184impl SnapshotSource for &LedgerSnapshot {
185    fn get(
186        &self,
187        key: &Rc<LedgerKey>,
188    ) -> Result<Option<(Rc<LedgerEntry>, Option<u32>)>, HostError> {
189        match self.ledger_entries.iter().find(|(k, _)| **k == **key) {
190            Some((_, v)) => Ok(Some((Rc::new(*v.0.clone()), v.1))),
191            None => Ok(None),
192        }
193    }
194}
