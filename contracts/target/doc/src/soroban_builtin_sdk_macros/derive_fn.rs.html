1use itertools::MultiUnzip;
2use proc_macro2::{Literal, TokenStream as TokenStream2};
3use quote::{format_ident, quote};
4use syn::{spanned::Spanned, Error, FnArg, Type};
5
6pub fn derive_contract_function_set<'a>(
7    ty: &Type,
8    methods: impl Iterator<Item = &'a syn::ImplItemFn>,
9) -> TokenStream2 {
10    let mut errors = Vec::<Error>::new();
11
12    let (str_lits, func_calls): (Vec<_>, Vec<_>) = methods
13        .enumerate()
14        .map(|(i, m)| {
15            let ident = &m.sig.ident;
16            let name = ident.to_string();
17            let str_lit = Literal::string(&name);
18            let (arg_indices, args, arg_types): (Vec<_>, Vec<_>, Vec<_>) = m.sig.inputs.iter().skip(1).cloned().enumerate().map(|(i, a)| {
19                let arg = format_ident!("arg{}", i);
20                match a {
21                    FnArg::Typed(t) => (i, arg, t.ty),
22                    _ => {
23                        errors.push(Error::new(a.span(), "invalid argument type"));
24                        (i, arg, syn::parse_quote! { () })
25                    }
26                }
27            }).multiunzip();
28            let num_args = args.len();
29            let func_call = quote! {
30                #i => {
31                    if args.len() == #num_args {
32                        #(let #args: #arg_types = args.get(#arg_indices).cloned().ok_or(soroban_env_common::ConversionError)?.try_into_val(host)?;)*
33                        Ok(Self::#ident(host, #(#args,)*)?.try_into_val(host)?)
34                    } else {
35                        Err(host.err(crate::xdr::ScErrorType::Context, crate::xdr::ScErrorCode::UnexpectedSize, "wrong number of args to func", &[func.into()]))
36                    }
37                }
38            };
39            (str_lit, func_call)
40        })
41        .multiunzip();
42
43    if !errors.is_empty() {
44        let compile_errors = errors.iter().map(Error::to_compile_error);
45        quote! { #(#compile_errors)* }
46    } else {
47        quote! {
48            impl crate::builtin_contracts::BuiltinContract for #ty {
49                fn call(
50                    &self,
51                    func: &soroban_env_common::Symbol,
52                    host: &crate::Host,
53                    args: &[soroban_env_common::Val],
54                ) -> Result<soroban_env_common::Val, crate::HostError> {
55                    use soroban_env_common::EnvBase;
56                    use super::*;
57                    const FNS: &'static [&'static str] = &[#(&#str_lits),*];
58                    match u32::from(host.symbol_index_in_strs(*func, FNS)?) as usize {
59                        #(#func_calls)*
60                        _ => Err(host.err(crate::xdr::ScErrorType::Context, crate::xdr::ScErrorCode::MissingValue, "function does not exist", &[func.into()]))
61                    }
62                }
63            }
64        }
65    }
66}
