6pub fn derive_type_struct(ident: &Ident, data: &DataStruct) -> TokenStream2 {
7    let len = Literal::usize_unsuffixed(data.fields.len());
8
9    let (idents, str_lits, idx_lits): (Vec<_>, Vec<_>, Vec<_>) =
10        if let Fields::Named(_) = &data.fields {
11            data.fields
12                .iter()
13                .sorted_by_key(|field| field.ident.as_ref().unwrap().to_string())
14                .enumerate()
15                .map(|(i, f)| {
16                    let ident = f.ident.as_ref().unwrap().clone();
17                    let str_lit = Literal::string(&ident.to_string());
18                    let idx_lit = Literal::usize_unsuffixed(i);
19                    (ident, str_lit, idx_lit)
20                })
21                .multiunzip()
22        } else {
23            data.fields
24                .iter()
25                .enumerate()
26                .map(|(i, _)| {
27                    let ident = format_ident!("{}", i);
28                    let str_lit = Literal::string(&ident.to_string());
29                    let idx_lit = Literal::usize_unsuffixed(i);
30                    (ident, str_lit, idx_lit)
31                })
32                .multiunzip()
33        };
34
35    quote! {
36
37        impl soroban_env_common::Compare<#ident> for crate::Host {
38            type Error = crate::HostError;
39            fn compare(&self, a: &#ident, b: &#ident) -> Result<core::cmp::Ordering, crate::HostError> {
40                #(match self.compare(&a.#idents, &b.#idents)? {
41                    core::cmp::Ordering::Equal => (),
42                    unequal => return Ok(unequal)
43                })*
44                Ok(core::cmp::Ordering::Equal)
45            }
46        }
47
48        impl soroban_env_common::TryFromVal<crate::Host, soroban_env_common::MapObject> for #ident {
49            type Error = crate::HostError;
50
51            fn try_from_val(env: &crate::Host, val: &soroban_env_common::MapObject) -> Result<Self, Self::Error> {
52                use soroban_env_common::EnvBase;
53                const KEYS: [&'static str; #len] = [#(#str_lits),*];
54                let mut vals: [soroban_env_common::Val; #len] = [soroban_env_common::Val::VOID.to_val(); #len];
55                env.map_unpack_to_slice(*val, &KEYS, &mut vals)?;
56                Ok(Self {
57                    #(#idents: vals[#idx_lits].try_into_val(env)?,)*
58                })
59            }
60        }
61
62        impl soroban_env_common::TryFromVal<crate::Host, soroban_env_common::Val> for #ident {
63            type Error = crate::HostError;
64
65            fn try_from_val(env: &crate::Host, val: &soroban_env_common::Val) -> Result<Self, Self::Error> {
66                let obj: soroban_env_common::MapObject = val.try_into()?;
67                obj.try_into_val(env)
68            }
69        }
70
71        impl soroban_env_common::TryFromVal<crate::Host, #ident> for soroban_env_common::Val {
72            type Error = crate::HostError;
73
74            fn try_from_val(env: &crate::Host, val: &#ident) -> Result<Self, Self::Error> {
75                use soroban_env_common::EnvBase;
76                const KEYS: [&'static str; #len] = [#(#str_lits),*];
77                let vals: [soroban_env_common::Val; #len] = [
78                    #(val.#idents.try_into_val(env)?),*
79                ];
80                Ok(env.map_new_from_slices(&KEYS, &vals)?.into())
81            }
82        }
83    }
84}

86pub fn derive_type_enum(ident: &Ident, data: &DataEnum) -> TokenStream2 {
87    let mut errors = Vec::<Error>::new();
88
89    let (str_lits, froms, intos, syms, compares): (Vec<_>, Vec<_>, Vec<_>, Vec<_>, Vec<_>) = data
90        .variants
91        .iter()
92        .enumerate()
93        .map(|(i, f)| {
94            let case_ident = &f.ident;
95            let case_name = case_ident.to_string();
96            let idx_lit = Literal::usize_unsuffixed(i);
97            let str_lit = Literal::string(&case_name);
98            let case_sym = quote! { crate::Symbol::try_from_val(env, &#str_lit) };
99
100            if f.fields.is_empty() {
101                let from = quote! {
102                    #idx_lit => Ok(Self::#case_ident)
103                };
104                let into = quote! {
105                    #ident::#case_ident => {
106                        Ok((#case_sym?,).try_into_val(env)?)
107                    }
108                };
109                let sym = quote! {
110                    #ident::#case_ident => #case_sym
111                };
112                let compare = quote! {
113                    (#ident::#case_ident, #ident::#case_ident) => Ok(core::cmp::Ordering::Equal)
114                };
115                (str_lit, from, into, sym, compare)
116            } else if f.fields.len() == 1 {
117                let from = quote! {
118                    #idx_lit => Ok(Self::#case_ident(vec.get(1)?))
119                };
120                let into = quote! {
121                    #ident::#case_ident(x) => {
122                        Ok((#case_sym?, crate::Val::try_from_val(env, x)?).try_into_val(env)?)
123                    }
124                };
125                let sym = quote! {
126                    #ident::#case_ident(_) => #case_sym
127                };
128                let compare = quote! {
129                    (#ident::#case_ident(a), #ident::#case_ident(b)) => self.compare(a, b)
130                };
131                (str_lit, from, into, sym, compare)
132            } else {
133                errors.push(Error::new(
134                    f.span(),
135                    "tuple variants with more than 1 element not supported",
136                ));
137                let from = quote! {};
138                let into = quote! {};
139                let cmp = quote! {};
140                let sym = quote! {};
141                (str_lit, from, into, cmp, sym)
142            }
143        })
144        .multiunzip();
145
146    if !errors.is_empty() {
147        let compile_errors = errors.iter().map(Error::to_compile_error);
148        quote! { #(#compile_errors)* }
149    } else {
150        quote! {
151
152            impl #ident {
153                fn discriminant_sym(&self, env: &crate::Host) -> Result<crate::Symbol, crate::Error> {
154                    use soroban_env_common::TryFromVal;
155                    match self {
156                        #(#syms,)*
157                    }
158                }
159            }
160
161            impl soroban_env_common::Compare<#ident> for crate::Host {
162                type Error = crate::HostError;
163                fn compare(&self, a: &#ident, b: &#ident) -> Result<core::cmp::Ordering, crate::HostError> {
164                    match (a, b) {
165                        #(#compares,)*
166                        _ => self.compare(&a.discriminant_sym(self)?, &b.discriminant_sym(self)?)
167                    }
168                }
169            }
170
171            impl soroban_env_common::TryFromVal<crate::Host, soroban_env_common::VecObject> for #ident {
172                type Error = crate::HostError;
173
174                fn try_from_val(env: &crate::Host, val: &soroban_env_common::VecObject) -> Result<Self, Self::Error> {
175                    use soroban_env_common::{EnvBase,TryFromVal};
176                    const CASES: &'static [&'static str] = &[#(#str_lits),*];
177                    let vec = crate::builtin_contracts::base_types::Vec::try_from_val(env, val)?;
178                    let discriminant: soroban_env_common::Symbol = vec.get(0)?;
179                    match u32::from(env.symbol_index_in_strs(discriminant, CASES)?) as usize {
180                        #(#froms,)*
181                        _ => Err(soroban_env_common::ConversionError.into())
182                    }
183                }
184            }
185
186            impl soroban_env_common::TryFromVal<crate::Host, soroban_env_common::Val> for #ident {
187                type Error = crate::HostError;
188
189                fn try_from_val(env: &crate::Host, val: &soroban_env_common::Val) -> Result<Self, Self::Error> {
190                    let obj: soroban_env_common::VecObject = val.try_into()?;
191                    obj.try_into_val(env)
192                }
193            }
194
195            impl soroban_env_common::TryFromVal<crate::Host, #ident> for soroban_env_common::Val {
196                type Error = crate::HostError;
197
198                fn try_from_val(env: &crate::Host, val: &#ident) -> Result<soroban_env_common::Val, Self::Error> {
199                    use soroban_env_common::TryFromVal;
200                    match val {
201                        #(#intos)*
202                    }
203                }
204            }
205        }
206    }
