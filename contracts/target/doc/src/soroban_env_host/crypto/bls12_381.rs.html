38fn units_of_fp<const EXPECTED_SIZE: usize>() -> u64 {
39    EXPECTED_SIZE.div_ceil(FP_SERIALIZED_SIZE) as u64
40}

44    // element types, which can be conceptually decomposed into units of Fp
45    // (the base field element), and will be charged accordingly.
46    // Validation of the deserialized entity must be performed outside of this
47    // function, to keep budget charging isolated.
48    pub(crate) fn deserialize_uncompressed_no_validate<
49        const EXPECTED_SIZE: usize,
50        T: CanonicalDeserialize,
51    >(
52        &self,
53        slice: &[u8],
54        tag: &str,
55    ) -> Result<T, HostError> {
56        if EXPECTED_SIZE == 0 || slice.len() != EXPECTED_SIZE {
57            return Err(self.err(
58                ScErrorType::Crypto,
59                ScErrorCode::InvalidInput,
60                format!("bls12-381 {tag}: invalid input length to deserialize").as_str(),
61                &[
62                    Val::from_u32(slice.len() as u32).into(),
63                    Val::from_u32(EXPECTED_SIZE as u32).into(),
64                ],
65            ));
66        }
67        self.as_budget().bulk_charge(
68            ContractCostType::Bls12381DecodeFp,
69            units_of_fp::<EXPECTED_SIZE>(),
70            None,
71        )?;
72        // validation turned off here to isolate the cost of serialization.
73        // proper validation has to be performed outside of this function
74        T::deserialize_with_mode(slice, Compress::No, Validate::No).map_err(|_e| {
75            self.err(
76                ScErrorType::Crypto,
77                ScErrorCode::InvalidInput,
78                format!("bls12-381: unable to deserialize {tag}").as_str(),
79                &[],
80            )
81        })
82    }

88        const EXPECTED_SIZE: usize,
89        T: CanonicalSerialize,
90    >(
91        &self,
92        element: &T,

99                ScErrorCode::InvalidInput,
100                format!("bls12-381 {tag}: invalid buffer length to serialize into").as_str(),
101                &[

123    fn validate_point_encoding<const EXPECTED_SIZE: usize>(
124        &self,
125        bytes: &[u8],
126        tag: &str,
127    ) -> Result<(), HostError> {
128        // validate input bytes length
129        if EXPECTED_SIZE == 0 || bytes.len() != EXPECTED_SIZE {
130            return Err(self.err(
131                ScErrorType::Crypto,
132                ScErrorCode::InvalidInput,
133                format!("bls12-381 {tag}: invalid input length to deserialize").as_str(),
134                &[
135                    Val::from_u32(bytes.len() as u32).into(),
136                    Val::from_u32(EXPECTED_SIZE as u32).into(),
137                ],
138            ));
139        }
140        // validated encoded flags. The most significant three bits encode the flags,
141        // i.e. `byte[0] == [compression_flag, infinity_flag, sort_flag, bit_3, .. bit_7]`
142        // - the compression_flag should be unset
143        // - the infinity_flag should be set **only if** rest of bits are all zero
144        // - the sort_flag should be unset
145        let flags = 0b1110_0000 & bytes[0];
146        match flags {
147            0b0100_0000 => {
148                // infinite bit is set, check all other bits are zero
149                let is_valid = bytes[0] == 0b0100_0000 && bytes[1..].iter().all(|x| x.is_zero());
150                if !is_valid {
151                    Err(self.err(ScErrorType::Crypto, ScErrorCode::InvalidInput, format!("bls12-381 {tag} deserialize: infinity flag (bit 1) is set while remaining bits are not all zero").as_str(), &[]))
152                } else {
153                    Ok(())
154                }
155            },
156            0b0000_0000 => Ok(()), // infinite bit is unset
157            _ => Err(self.err(ScErrorType::Crypto, ScErrorCode::InvalidInput, format!("bls12-381 {tag} deserialize: either compression flag (bit 0) or the sort flag (bit 2) is set, while the input should be encoded uncompressed").as_str(), &[]))
158        }
159    }

181    pub(crate) fn affine_deserialize<const EXPECTED_SIZE: usize, P: SWCurveConfig>(
182        &self,
183        bo: BytesObject,
184        ct_curve: ContractCostType,
185        subgroup_check: bool,
186        ct_subgroup: ContractCostType,
187        tag: &str,
188    ) -> Result<Affine<P>, HostError> {
189        let pt: Affine<P> = self.visit_obj(bo, |bytes: &ScBytes| {
190            self.validate_point_encoding::<EXPECTED_SIZE>(&bytes, tag)?;
191            // `CanonicalDeserialize` of `Affine<P>` calls into
192            // `P::deserialize_with_mode`, where `P` is `arc_bls12_381::{g1,g2}::Config`, the
193            // core logic is in `arc_bls12_381::curves::util::read_{g1,g2}_uncompressed`.
194            //
195            // The `arc_bls12_381` lib already expects the input to be serialized in
196            // big-endian order (aligning with the common standard and contrary
197            // to ark::serialize's convention),
198            //
199            // i.e. `input = be_bytes(X) || be_bytes(Y)` and the
200            // most-significant three bits of X are flags:
201            //
202            // `bits(Affine) = [compression_flag, infinity_flag, sort_flag, ..remaining X_bits.., ..Y_bits..]`
203            //
204            // For `G1Affine`, each coordinate is an `Fp` that is 48 bytes.
205            //
206            // For `G2Affine`, each coordinate is an `Fp2` which contains two `Fp`,
207            // i.e. `(c1: Fp, c0: Fp)` see `field_element_deserialize` for more details.
208            //
209            // Internally when deserializing `Fp`, the flag bits are masked off
210            // to get `X: Fp`. The Y however, does not have the top bits masked off
211            // so it is possible for Y to exceed 381 bits. Internally `Fp` deserialization
212            // makes sure any value >= prime modulus results in an error.
213            self.deserialize_uncompressed_no_validate::<EXPECTED_SIZE, _>(bytes.as_slice(), tag)
214        })?;
215        if !self.check_point_is_on_curve(&pt, &ct_curve)? {
216            return Err(self.err(
217                ScErrorType::Crypto,
218                ScErrorCode::InvalidInput,
219                format!("bls12-381 {}: point not on curve", tag).as_str(),
220                &[],
221            ));
222        }
223        if subgroup_check && !self.check_point_is_in_subgroup(&pt, &ct_subgroup)? {
224            return Err(self.err(
225                ScErrorType::Crypto,
226                ScErrorCode::InvalidInput,
227                format!("bls12-381 {}: point not in the correct subgroup", tag).as_str(),
228                &[],
229            ));
230        }
231        Ok(pt)
232    }

332        self.charge_budget(ContractCostType::Bls12381FrFromU256, None)?;
333        let fr = if let Ok(small) = U256Small::try_from(sv) {

337            self.visit_obj(obj, |u: &U256| {

339            })?
340        };
341        Ok(fr)
342    }

344    pub(crate) fn fr_to_u256val(&self, scalar: Fr) -> Result<U256Val, HostError> {
345        self.charge_budget(ContractCostType::Bls12381FrToU256, None)?;
346        // The `into_bigint` carries the majority of the cost. It performs the
347        // Montgomery reduction on the internal representation, which is doing a
348        // number of wrapping arithmetics on each u64 word (`Fr` contains 4
349        // words). The core routine is in `ark_ff::MontConfig::into_bigint`,
350        // this cannot panic.
351        let bytes: [u8; 32] = scalar
352            .into_bigint()
353            .to_bytes_be()
354            .try_into()
355            .map_err(|_| HostError::from(ConversionError))?;
356        let u = U256::from_be_bytes(bytes);
357        self.map_err(U256Val::try_from_val(self, &u))
358    }

360    pub(crate) fn field_element_deserialize<const EXPECTED_SIZE: usize, T: CanonicalDeserialize>(
361        &self,
362        bo: BytesObject,
363        tag: &str,
364    ) -> Result<T, HostError> {
365        self.visit_obj(bo, |bytes: &ScBytes| {
366            if bytes.len() != EXPECTED_SIZE {
367                return Err(self.err(
368                    ScErrorType::Crypto,
369                    ScErrorCode::InvalidInput,
370                    format!(
371                        "bls12-381 field element {}: invalid input length to deserialize",
372                        tag
373                    )
374                    .as_str(),
375                    &[
376                        Val::from_u32(bytes.len() as u32).into(),
377                        Val::from_u32(EXPECTED_SIZE as u32).into(),
378                    ],
379                ));
380            }
381            self.charge_budget(ContractCostType::MemCpy, Some(EXPECTED_SIZE as u64))?;
382            let mut buf = [0u8; EXPECTED_SIZE];
383            buf.copy_from_slice(bytes);
384            buf.reverse();
385
386            // The field element here an either be a Fp<P, N=6> (base field
387            // element) or QuadExtField<P> (quadratic extension)
388            //
389            // - `CanonicalDeserialize for Fp<P, N>` assumes input bytes in
390            // little-endian order, with the highest (right-most) bits being
391            // empty flags. This is reverse of our rule, which assumes
392            // big-endian order with the highest (left-most) bits for flags.
393            //
394            // - `CanonicalDeserialize for QuadExtField<P>` reads the first
395            // chunk, deserialize it into `Fp` as `c0`. Then repeat for `c1`. The
396            // deserialization for `Fp` follows same rules as above, where the
397            // bytes are expected in little-endian, with the highest bits being
398            // empty flags. There is no check involved. This is entirely
399            // reversed from our input format: `be_bytes(c1) || be_bytes(c0)` from
400            // [standard](https://github.com/zcash/librustzcash/blob/6e0364cd42a2b3d2b958a54771ef51a8db79dd29/pairing/src/bls12_381/README.md#serialization)
401            //
402            // In either case, we just need to reverse the input bytes before
403            // passing them in. There is no other check for `Fp` besides the
404            // length check, internally it makes sure `Fp` is valid integer
405            // modulo `q` (the prime modulus)
406            self.deserialize_uncompressed_no_validate::<EXPECTED_SIZE, _>(&buf, tag)
407        })
408    }

419        let len: u32 = self.vec_len(vs)?.into();
420        let mut scalars: Vec<Fr> = vec![];

528    pub(crate) fn msm_internal<P: SWCurveConfig>(
529        &self,
530        points: &[Affine<P>],
531        scalars: &[<P as CurveConfig>::ScalarField],
532        ty: &ContractCostType,
533        tag: &str,
534    ) -> Result<Projective<P>, HostError> {
535        self.charge_budget(*ty, Some(points.len() as u64))?;
536        if points.len() != scalars.len() || points.len() == 0 {
537            return Err(self.err(
538                ScErrorType::Crypto,
539                ScErrorCode::InvalidInput,
540                format!(
541                    "{tag} msm: invalid input vector lengths ({}, {})",
542                    points.len(),
543                    scalars.len()
544                )
545                .as_str(),
546                &[],
547            ));
548        }
549        // The actual logic happens inside msm_bigint_wnaf (ark_ec/variable_base/mod.rs)
550        // under branch negation is cheap.
551        // the unchecked version just skips the length equal check
552        Ok(Projective::<P>::msm_unchecked(points, scalars))
553    }

555    pub(crate) fn map_to_curve<P: WBConfig>(
556        &self,
557        fp: <Affine<P> as AffineRepr>::BaseField,
558        ty: ContractCostType,
559    ) -> Result<Affine<P>, HostError> {
560        self.charge_budget(ty, None)?;
561
562        // The `WBMap<g2::Config>::new()` first calls
563        // `P::ISOGENY_MAP.apply(GENERATOR)` which returns error if the result
564        // point is not on curve. This should not happen if the map constants
565        // have been correctly defined. otherwise it would be an internal error
566        // since it's a bug in the library implementation.
567        //
568        // Then it returns `WBMap`, which wraps a `SWUMap<P>` where P is the
569        // `ark_bls12_381::curves::g2_swu_iso::SwuIsoConfig`.
570        //
571        // Potential panic condition: `SWUMap::new().unwrap()`
572        //
573        // The `SWUMap::new()` function performs some validation on the static
574        // parameters `ZETA`, `COEFF_A`, `COEFF_B`, all of which are statically
575        // defined in `ark_bls12_381::curves::g1_swu_iso` and `g2_swu_iso`.
576        // Realistically this panic cannot occur, otherwise it will panic every
577        // time including during tests
578        let mapper = WBMap::<P>::new().map_err(|e| {
579            self.err(
580                ScErrorType::Crypto,
581                ScErrorCode::InternalError,
582                format!("hash-to-curve error {e}").as_str(),
583                &[],
584            )
585        })?;
586
587        // The `SWUMap::map_to_curve` function contains several panic conditions
588        // 1. assert!(!div3.is_zero())
589        // 2. gx1.sqrt().expect()
590        // 3. zeta_gx1.sqrt().expect()
591        // 4. assert!(point_on_curve.is_on_curve())
592        //
593        // While all of these should theoretically just be debug assertions that
594        // can't happen if the map parameters are correctly defined (several of
595        // these have recently been downgraded to debug_assert, e.g see
596        // https://github.com/arkworks-rs/algebra/pull/659#discussion_r1450808159),
597        // we cannot guaruantee with 100% confidence these panics will never
598        // happen.
599        //
600        // Otherwise, this function should never Err.
601        mapper.map_to_curve(fp).map_err(|e| {
602            self.err(
603                ScErrorType::Crypto,
604                ScErrorCode::InternalError,
605                format!("hash-to-curve error {e}").as_str(),
606                &[],
607            )
608        })
609    }

611    pub(crate) fn hash_to_curve<P: WBConfig>(
612        &self,
613        domain: &[u8],
614        msg: &[u8],
615        ty: &ContractCostType,
616    ) -> Result<Affine<P>, HostError> {
617        self.charge_budget(*ty, Some(msg.len() as u64))?;
618        // check dst requirements
619        let dst_len = domain.len();
620        if dst_len == 0 || dst_len > 255 {
621            return Err(self.err(
622                ScErrorType::Crypto,
623                ScErrorCode::InvalidInput,
624                format!("hash_to_curve: invalid input dst length {dst_len}, must be > 0 and < 256")
625                    .as_str(),
626                &[],
627            ));
628        }
629
630        // The `new` function here constructs a DefaultFieldHasher and a WBMap.
631        // - The DefaultFieldHasher::new() function creates an ExpanderXmd with
632        // Sha256. This cannot fail or panic.
633        // - Construction of WBMap follows the exact same analysis as map_to_curve
634        // function earlier.
635        // This function cannot realistically produce an error or panic.
636        let mapper =
637            MapToCurveBasedHasher::<Projective<P>, DefaultFieldHasher<Sha256, 128>, WBMap<P>>::new(
638                domain,
639            )
640            .map_err(|e| {
641                self.err(
642                    ScErrorType::Crypto,
643                    ScErrorCode::InternalError,
644                    format!("hash-to-curve error {e}").as_str(),
645                    &[],
646                )
647            })?;
648
649        // `ark_ec::hashing::map_to_curve_hasher::MapToCurveBasedHasher::hash`
650        // contains the following calls
651        // - `DefaultFieldHasher::hash_to_field`
652        // - `SWUMap::map_to_curve`
653        // - `clear_cofactor`. This cannot fail or panic.
654        //
655        // `hash_to_field` calls the ExpanderXmd::expand function, there are two
656        // assertions on the length of bytes produced by the hash function. Both
657        // of these cannot happen because the output size can be computed
658        // analytically. Let's use G2:
659        // - `block_size = 384 (Fp bit size) + 128 (security padding) / 8 = 64`
660        // - `len_in_bytes = 2 (number of elements to produce) *  2 (extention
661        //   degree of Fp2) * 64 (block_size) = 256`
662        // - `ell = 256 (len_in_bytes) / 32 (sha256 output size) = 8`
663        //
664        // # Assertion #1. ell <= 255, which is saying the expander cannot expand
665        // up to a certain length. in our case ell == 8.
666        // # Assertion #2. len_in_bytes < 2^16, which is clearly true as well.
667        //
668        // The rest is just hashing, dividing bytes into element size, and
669        // producing field elements from bytes. None of these can panic or
670        // error.
671        //
672        // The only panic conditions we cannot 100% exclude comes from
673        // `map_to_curve`, see previous analysis.
674        //
675        // This function should not Err.
676        mapper.hash(msg.as_ref()).map_err(|e| {
677            self.err(
678                ScErrorType::Crypto,
679                ScErrorCode::InternalError,
680                format!("hash-to-curve error {e}").as_str(),
681                &[],
682            )
683        })
684    }

686    pub(crate) fn pairing_internal(
687        &self,
688        vp1: &Vec<G1Affine>,
689        vp2: &Vec<G2Affine>,
690    ) -> Result<PairingOutput<Bls12_381>, HostError> {
691        self.charge_budget(ContractCostType::Bls12381Pairing, Some(vp1.len() as u64))?;
692        // check length requirements
693        if vp1.len() != vp2.len() || vp1.len() == 0 {
694            return Err(self.err(
695                ScErrorType::Crypto,
696                ScErrorCode::InvalidInput,
697                format!(
698                    "pairing: invalid input vector lengths ({}, {})",
699                    vp1.len(),
700                    vp2.len()
701                )
702                .as_str(),
703                &[],
704            ));
705        }
706
707        // This calls into `Bls12<Config>::multi_miller_loop`, which just calls
708        // `ark_ec::models::bls12::Bls12Config::multi_miller_loop` with specific
709        // parameters defined in `ark_bls12_381::curves`.
710        //
711        // Panic analysis:
712        //
713        // The following potential panic conditions could exist:
714        // 1. if two input vector lengths are not equal. There is a `zip_eq`
715        // which panics if the length of the two vectors are not equal. This is
716        // weeded out up front.
717        //
718        // 2. `coeffs.next().unwrap()`. This occurs when the algorithm Loops
719        // over pairs of `(a: G1Affine, b: G2Affine)`, converting them into
720        // `Vec<(G1Prepared, G2Preared::EllCoeff<Config>)>`, the latter contains
721        // three elements of Fp2. For each pair, the coeffs.next() can at most
722        // be called twice, when the bit being looped over in `Config::X` is
723        // set. So this panic cannot happen.
724        //
725        // 3. if any of the G1Affine point is infinity. The ell() function which
726        // calls p.xy().unwrap(), which is when the point is infinity. This
727        // condition also cannot happen because when the pairs are generated,
728        // any pair containing a zero point is filtered.
729        //
730        // The above analysis is best effort to weed out panics from the source,
731        // however the algorithm is quite involved. So we cannot be 100% certain
732        // every panic condition has been excluded.
733        let mlo = Bls12_381::multi_miller_loop(vp1, vp2);
734        // final_exponentiation returning None means the `mlo.0.is_zero()`
735        Bls12_381::final_exponentiation(mlo).ok_or_else(|| {
736            self.err(
737                ScErrorType::Crypto,
738                ScErrorCode::InvalidInput,
739                "final_exponentiation has failed, most likely multi_miller_loop produced infinity",
740                &[],
741            )
742        })
743    }

746        &self,
747        output: &PairingOutput<Bls12_381>,
748    ) -> Result<Bool, HostError> {
749        self.charge_budget(ContractCostType::MemCmp, Some(FP12_SERIALIZED_SIZE as u64))?;
750        match output.0.cmp(&Fq12::ONE) {
751            Ordering::Equal => Ok(true.into()),
752            _ => Ok(false.into()),
753        }
754    }

782    pub(crate) fn fr_inv_internal(&self, lhs: &Fr) -> Result<Fr, HostError> {
783        if lhs.is_zero() {
784            return Err(self.err(
785                ScErrorType::Crypto,
786                ScErrorCode::InvalidInput,
787                "scalar inversion input is zero",
788                &[],
789            ));
790        }
791        self.charge_budget(ContractCostType::Bls12381FrInv, None)?;
792        // `inverse()` returns `None` only if the rhs is zero, which we have
793        // checked upfront, so this cannot fail.
794        lhs.inverse().ok_or_else(|| {
795            self.err(
796                ScErrorType::Crypto,
797                ScErrorCode::InternalError,
798                "scalar inversion failed",
799                &[],
800            )
801        })
802    }
