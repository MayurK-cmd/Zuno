15pub(crate) fn read_allowance(e: &Host, from: Address, spender: Address) -> Result<i128, HostError> {
16    let key = DataKey::Allowance(AllowanceDataKey { from, spender });
17    if let Some(allowance) =
18        e.try_get_contract_data(key.try_into_val(e)?, StorageType::Temporary)?
19    {
20        let val: AllowanceValue = allowance.try_into_val(e)?;
21        if val.live_until_ledger < e.get_ledger_sequence()?.into() {
22            Ok(0)
23        } else {
24            Ok(val.amount)
25        }
26    } else {
27        Ok(0)
28    }
29}

31// Metering: covered by components
32pub(crate) fn write_allowance(
33    e: &Host,
34    from: Address,
35    spender: Address,
36    amount: i128,
37    live_until: u32,
38) -> Result<(), HostError> {
39    let key = DataKey::Allowance(AllowanceDataKey { from, spender });
40
41    // Validates live_until and then returns the ledger seq
42    // The live_until can be less than ledger seq if clearing an allowance
43    let ledger_seq = e.with_ledger_info(|li| {
44        if live_until > e.max_live_until_ledger()? {
45            Err(err!(
46                e,
47                ContractError::AllowanceError,
48                "live_until is greater than max",
49                live_until,
50                li.max_entry_ttl
51            ))
52        } else if amount > 0 && live_until < li.sequence_number {
53            Err(err!(
54                e,
55                ContractError::AllowanceError,
56                "live_until must be >= ledger sequence",
57                live_until,
58                li.sequence_number
59            ))
60        } else {
61            Ok(li.sequence_number)
62        }
63    })?;
64
65    // Returns the allowance to write and the previous live_until ledger of the existing allowance.
66    // If an allowance didn't exist, then the previous live_until ledger will be None.
67    let allowance_with_live_until_option: Option<(AllowanceValue, Option<u32>)> =
68        if let Some(allowance) =
69            e.try_get_contract_data(key.try_into_val(e)?, StorageType::Temporary)?
70        {
71            let mut updated_allowance: AllowanceValue = allowance.try_into_val(e)?;
72            updated_allowance.amount = amount;
73
74            let old_live_until = updated_allowance.live_until_ledger;
75            updated_allowance.live_until_ledger = live_until;
76            Some((updated_allowance, Some(old_live_until)))
77        } else if amount > 0 {
78            Some((
79                AllowanceValue {
80                    amount,
81                    live_until_ledger: live_until,
82                },
83                None,
84            ))
85        } else {
86            None
87        };
88
89    match allowance_with_live_until_option {
90        Some(allowance_with_live_until) => {
91            e.put_contract_data(
92                key.try_into_val(e)?,
93                allowance_with_live_until.0.try_into_val(e)?,
94                StorageType::Temporary,
95            )?;
96
97            if allowance_with_live_until.0.amount > 0
98                && allowance_with_live_until.1.unwrap_or(0) < live_until
99            {
100                let live_for = live_until.saturating_sub(ledger_seq).saturating_add(1);
101                e.extend_contract_data_ttl(
102                    key.try_into_val(e)?,
103                    StorageType::Temporary,
104                    live_for.into(),
105                    live_for.into(),
106                )?;
107            }
108        }
109        None => {}
110    }
111
112    Ok(())
113}

115// allowance is expected to exist
116fn write_allowance_amount(
117    e: &Host,
118    from: Address,
119    spender: Address,
120    amount: i128,
121) -> Result<(), HostError> {
122    let key = DataKey::Allowance(AllowanceDataKey {
123        from: from.metered_clone(e)?,
124        spender: spender.metered_clone(e)?,
125    });
126
127    let allowance: AllowanceValue = e
128        .get_contract_data(key.try_into_val(e)?, StorageType::Temporary)?
129        .try_into_val(e)?;
130    write_allowance(e, from, spender, amount, allowance.live_until_ledger)
131}

133// Metering: covered by components
134pub(crate) fn spend_allowance(
135    e: &Host,
136    from: Address,
137    spender: Address,
138    amount: i128,
139) -> Result<(), HostError> {
140    let allowance = read_allowance(e, from.metered_clone(e)?, spender.metered_clone(e)?)?;
141    if allowance < amount {
142        return Err(err!(
143            e,
144            ContractError::AllowanceError,
145            "not enough allowance to spend",
146            allowance,
147            amount
148        ));
149    } else if amount > 0 {
150        let new_allowance = allowance.checked_sub(amount).ok_or_else(|| {
151            e.error(
152                ContractError::OverflowError.into(),
153                "allowance overflowed",
154                &[],
155            )
156        })?;
157        write_allowance_amount(e, from, spender, new_allowance)?;
158    }
159    Ok(())
160}
