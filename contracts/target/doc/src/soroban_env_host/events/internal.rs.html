11/// The internal representation of a `ContractEvent` that is stored in the
12/// events buffer and designed to be cheap to clone.
13// This is exposed as a pub type for benches.
14#[derive(Clone, Debug)]
15pub(crate) struct InternalContractEvent {
16    pub type_: xdr::ContractEventType,
17    pub contract_id: Option<BytesObject>,
18    pub topics: VecObject,
19    pub data: Val,
20}
21
22impl InternalContractEvent {
23    // Metering: covered by components
24    fn to_xdr(&self, host: &Host) -> Result<xdr::ContractEvent, HostError> {
25        let topics = host.vecobject_to_scval_vec(self.topics)?;
26        let data = host.from_host_val(self.data)?;
27        let contract_id = match self.contract_id {
28            Some(id) => Some(host.hash_from_bytesobj_input("contract_id", id)?),
29            None => None,
30        };
31        Ok(xdr::ContractEvent {
32            ext: xdr::ExtensionPoint::V0,
33            contract_id,
34            type_: self.type_,
35            body: xdr::ContractEventBody::V0(xdr::ContractEventV0 { topics, data }),
36        })
37    }
38}

40/// Internal representation of a structured debug message which is logically
41/// similar to an InternalContractEvent (and will project to a ContractEvent the
42/// same way when we externalize it to XDR) but internally is different: it
43/// stores more "plain" rust data values -- Vec as well as XDR types like ScVal
44/// -- capturing the input to diagnostics, rather than interacting with the host
45/// object system. It does this for two reasons:
46///
47///   1. to avoid perturbing object numbering when debug-mode is
48///      enabled -- this could potentially cause hosts running on watcher nodes
49///      with debug enabled to diverge from validators with debug disabled.
50///
51///   2. to avoid re-entering the host object management code by adding (say)
52///      host objects for the contract ID, topics vector or such in the middle
53///      of a diagnostic call occurs during a read from the host object table
54///      (eg. validating a host function call's input arguments).
55
56#[derive(Clone, Debug)]
57pub(crate) struct InternalDiagnosticEvent {
58    pub contract_id: Option<crate::xdr::Hash>,
59    pub topics: Vec<InternalDiagnosticArg>,
60    pub args: Vec<InternalDiagnosticArg>,
61}
62
63impl std::hash::Hash for InternalEvent {
64    fn hash<H: std::hash::Hasher>(&self, state: &mut H) {
65        match self {
66            InternalEvent::Contract(c) => {
67                c.type_.hash(state);
68                match c.contract_id {
69                    Some(cid) => cid.to_val().get_payload().hash(state),
70                    None => 0.hash(state),
71                }
72                c.data.get_payload().hash(state);
73                c.topics.to_val().get_payload().hash(state);
74            }
75            // These are not included in the hash because they not supposed to
76            // be observable, and we only have hashing to support
77            // test-observation.
78            InternalEvent::Diagnostic(_) => (),
79        }
80    }
81}
82
83impl std::hash::Hash for EventError {
84    fn hash<H: std::hash::Hasher>(&self, state: &mut H) {
85        match self {
86            EventError::FromFailedCall => 0.hash(state),
87            EventError::FromSuccessfulCall => 1.hash(state),
88        }
89    }
90}

92// As mentioned above, we want to support storing "plain" rust datatypes as
93// arguments to diagnostic events (in the form of ScVals), but in certain cases
94// the user will be providing an _existing_ host object reference in which case
95// we allow storing that as an argument too. We avoid eagerly converting it to
96// an ScVal to avoid wasting CPU in the standard case where nobody is going to
97// observe the event anyway.
98#[derive(Clone, Debug)]
99pub(crate) enum InternalDiagnosticArg {
100    HostVal(Val),
101    XdrVal(ScVal),
102}
103
104fn externalize_args(host: &Host, args: &[InternalDiagnosticArg]) -> Result<Vec<ScVal>, HostError> {
105    if !host.as_budget().is_in_shadow_mode()? {
106        return Err(host.err(
107            xdr::ScErrorType::Events,
108            xdr::ScErrorCode::InternalError,
109            "`externalize_args` can only be called in debug mode",
110            &[],
111        ));
112    }
113
114    args.iter()
115        .map(|arg| match arg {
116            InternalDiagnosticArg::HostVal(h) => host.from_host_val(*h),
117            InternalDiagnosticArg::XdrVal(v) => v.metered_clone(host),
118        })
119        .metered_collect::<Result<Vec<ScVal>, HostError>>(host)?
120}
121
122impl InternalDiagnosticEvent {
123    fn to_xdr(&self, host: &Host) -> Result<xdr::ContractEvent, HostError> {
124        if !host.as_budget().is_in_shadow_mode()? {
125            return Err(host.err(
126                xdr::ScErrorType::Events,
127                xdr::ScErrorCode::InternalError,
128                "`InternalDiagnosticEvent::to_xdr` can only be called in debug mode",
129                &[],
130            ));
131        }
132
133        let topics: xdr::VecM<ScVal> = externalize_args(host, &self.topics)?.try_into()?;
134        let args = externalize_args(host, &self.args)?;
135        let data = if args.len() > 1 {
136            ScVal::Vec(Some(xdr::ScVec::from(xdr::VecM::try_from(args)?)))
137        } else if let Some(arg) = args.into_iter().next() {
138            arg
139        } else {
140            ScVal::Void
141        };
142        Ok(xdr::ContractEvent {
143            ext: xdr::ExtensionPoint::V0,
144            contract_id: self.contract_id.metered_clone(host)?,
145            type_: xdr::ContractEventType::Diagnostic,
146            body: xdr::ContractEventBody::V0(xdr::ContractEventV0 { topics, data }),
147        })
148    }
149}

151/// The internal representation of an `Event` that is stored in the events buffer
152/// and designed to be cheap to clone.
153#[derive(Clone, Debug)]
154pub(crate) enum InternalEvent {
155    Contract(InternalContractEvent),
156    Diagnostic(Rc<InternalDiagnosticEvent>),
157}
158
159#[derive(Clone, Debug, PartialEq)]
160pub(crate) enum EventError {
161    FromFailedCall,
162    FromSuccessfulCall,
163}
164
165/// The events buffer. Stores `InternalEvent`s in the chronological order.
166#[derive(Clone, Default)]
167pub(crate) struct InternalEventsBuffer {
168    pub(crate) vec: Vec<(InternalEvent, EventError)>,
169}
170
171impl InternalEventsBuffer {
172    // Records an InternalEvent
173    pub(crate) fn record(&mut self, e: InternalEvent, host: &Host) -> Result<(), HostError> {
174        let mut metered_internal_event_push = |e: InternalEvent| -> Result<(), HostError> {
175            // Metering: we use the cost of instantiating a size=1 `Vec` as an
176            // estimate for the cost `Vec.push(event)`.  Because the buffer length
177            // may be different on different instances due to diagnostic events
178            // and we need a deterministic cost across all instances, the cost
179            // needs to be amortized and buffer size-independent.
180            Vec::<(InternalEvent, EventError)>::charge_bulk_init_cpy(1, host)?;
181            self.vec.push((e, EventError::FromSuccessfulCall));
182            Ok(())
183        };
184
185        match &e {
186            InternalEvent::Contract(_) => metered_internal_event_push(e)?,
187            InternalEvent::Diagnostic(_) => host.with_debug_mode(|| metered_internal_event_push(e)),
188        }
189
190        Ok(())
191    }
192
193    /// "Rolls back" the event buffer starting at `events` by marking all
194    /// subsequent events as failed calls.
195    pub(crate) fn rollback(&mut self, events: usize) -> Result<(), HostError> {
196        // note that we first skip the events that are not being rolled back
197        // Metering: free (or conceptually: paid for when pushing the event)
198        for e in self.vec.iter_mut().skip(events) {
199            e.1 = EventError::FromFailedCall;
200        }
201
202        Ok(())
203    }
204
205    /// Converts the internal events into their external representation. This
206    /// should only be called either when the host is finished (via
207    /// `try_finish`), or when an error occurs.
208    pub(crate) fn externalize(&self, host: &Host) -> Result<Events, HostError> {
209        // This line is intentionally unmetered. We want to separate out
210        // charging the main budget for `Contract` events (with "observable"
211        // costs) from charging the debug budget for `Diagnostic` events (with
212        // "non-observable" costs). Both event types are stored in the same
213        // input vector so we must not do a bulk charge based on its length, but
214        // rather walk through it charging to one budget or the other on an
215        // event-by-event basis.
216        let mut vec = Vec::with_capacity(self.vec.len());
217
218        let mut metered_external_event_push =
219            |event: xdr::ContractEvent, status: &EventError| -> Result<(), HostError> {
220                // Metering: we use the cost of instantiating a size=1 `Vec` as
221                // an estimate for the cost collecting 1 `HostEvent` into the
222                // events buffer. Because the resulting buffer length may be
223                // different on different instances (due to diagnostic events)
224                // and we need a deterministic cost across all instances, the
225                // cost needs to be amortized and buffer size-independent.
226                Vec::<HostEvent>::charge_bulk_init_cpy(1, host)?;
227                vec.push(HostEvent {
228                    event,
229                    failed_call: *status == EventError::FromFailedCall,
230                });
231                Ok(())
232            };
233
234        for (event, status) in self.vec.iter() {
235            match event {
236                InternalEvent::Contract(c) => {
237                    metered_external_event_push(c.to_xdr(host)?, status)?;
238                }
239                InternalEvent::Diagnostic(d) => {
240                    host.with_debug_mode(|| metered_external_event_push(d.to_xdr(host)?, status));
241                }
242            }
243        }
244
245        Ok(Events(vec))
246    }
247
248    /// Converts the internal diagnostic events into their external representation.
249    #[cfg(any(test, feature = "testutils"))]
250    pub(crate) fn externalize_diagnostics(&self, host: &Host) -> Result<Events, HostError> {
251        let mut vec = Vec::with_capacity(self.vec.len());
252
253        for (event, status) in self.vec.iter() {
254            match event {
255                InternalEvent::Contract(_) => {}
256                InternalEvent::Diagnostic(d) => {
257                    host.with_debug_mode(|| {
258                        vec.push(HostEvent {
259                            event: d.to_xdr(host)?,
260                            failed_call: *status == EventError::FromFailedCall,
261                        });
262                        Ok(())
263                    });
264                }
265            }
266        }
267
268        Ok(Events(vec))
269    }
270
271    #[cfg(any(test, feature = "testutils"))]
272    pub(crate) fn clear(&mut self) {
273        self.vec.clear();
274    }
