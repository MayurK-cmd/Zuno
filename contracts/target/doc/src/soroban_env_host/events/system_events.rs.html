1use crate::{builtin_contracts::common_types::ContractExecutable, Host, HostError};
2use soroban_env_common::{xdr, EnvBase, Symbol, TryFromVal, TryIntoVal, Val, VecObject};
3
4const CONTRACT_EXECUTABLE_UPDATE_TOPIC: &str = "executable_update";
5
6impl Host {
7    pub(crate) fn system_event(&self, topics: VecObject, data: Val) -> Result<(), HostError> {
8        self.record_contract_event(xdr::ContractEventType::System, topics, data)?;
9        Ok(())
10    }
11
12    // Emits a system event for updating the contract executable. The topic
13    // vector contains the symbol "executable_update" followed by the
14    // old_executable and new_executable, encoded as contract types
15    // (`ContractExecutable` above). The event data is empty.
16    pub(crate) fn emit_update_contract_event(
17        &self,
18        old_executable: &xdr::ContractExecutable,
19        new_executable: &xdr::ContractExecutable,
20    ) -> Result<(), HostError> {
21        self.system_event(
22            self.vec_new_from_slice(&[
23                Symbol::try_from_val(self, &CONTRACT_EXECUTABLE_UPDATE_TOPIC)?.into(),
24                ContractExecutable::from_xdr(self, old_executable)?.try_into_val(self)?,
25                ContractExecutable::from_xdr(self, new_executable)?.try_into_val(self)?,
26            ])?,
27            self.vec_new_from_slice(&[])?.into(),
28        )?;
29        Ok(())
30    }
31}
