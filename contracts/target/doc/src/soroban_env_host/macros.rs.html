1// This module contains ubiquitous macros that get used in many other modules
2// in the host crate. This module is ordered before any other module in the
3// crate to ensure availability of the macros using #[macro_use]. The other
4// option is to add weird `pub(crate) use foo` statements after each definition
5// and import them as items, which is a more modern style, but we avoid it here
6// and use the older style for simplicity.

28#[cfg(test)]
29macro_rules! function_name {
30    () => {{
31        // This is the canonical hack used to get a function's name,
32        // copied from the stdext crate (and stackoverflow).
33        fn f() {}
34        fn type_name_of<T>(_: T) -> &'static str {
35            std::any::type_name::<T>()
36        }
37        let name = type_name_of(f);
38        // `3` is the length of the `::f`.
39        &name[..name.len() - 3]
40    }};
41}
42
43macro_rules! function_short_name {
44    () => {{
45        fn f() {}
46        fn type_name_of<T>(_: T) -> &'static str {
47            std::any::type_name::<T>()
48        }
49        let name = type_name_of(f);
50        // `3` is the length of the `::f`.
51        let name = &name[..name.len() - 3];
52        &name[name.rfind("::").map(|x| x + 2).unwrap_or(0)..]
53    }};
54}
