1//! Functions to support measurement of the cost of cloning (and allocating)
2//! objects that may contain pointers into the heap.
3//!
4//! Note that in this file the term "substructure" refers to the presence of
5//! _pointers to owned data_ in a type, such that "merely counting the size" of
6//! the type fails to capture the cost of cloning it. For example a struct with
7//! 10 `Option<i32>` fields has "no substructure", because counting the size of
8//! the struct gives you the right cost of cloning it; whereas a struct with
9//! even one `Vec<u8>` is considered to "have substructure" because the struct's
10//! size is just the fixed size of the vector header, and fails to account to
11//! the owned heap allocation that the `Vec` _points to_, and that will be
12//! cloned anytime we clone the struct.
13//!
14//! Consequently in methods where we "charge for the substructure" we are only
15//! concerned with visiting any fields that (transitively) contain owning
16//! pointers. The rest of each struct is considered its "shallow" part and
17//! charged based on its declared size (see [`DeclaredSizeForMetering`]).
18

45/// Charge for an N-element "shallow copy" of some type, not cloning any substructure. The charge
46/// unit is number of elements `n_elts` multiplied by a declared size of each element. In a better
47/// world we would multiply by `size_of<Self>` instead but that's not guaranteed to be stable, which
48/// might cause metering to differ across compilations, causing problems in concensus and replay.
49pub(crate) fn charge_shallow_copy<T: DeclaredSizeForMetering>(
50    n_elts: u64,
51    budget: impl AsBudget,
52) -> Result<(), HostError> {
53    // Ideally we would want a static assertion. However, it does not work due to rust restrictions
54    // (e.g. see https://github.com/rust-lang/rust/issues/57775). Here we make a runtime assertion
55    // that the type's size is below its promised element size for budget charging. This assertion
56    // only happens in debug build. In optimized build, not satisfying the asserted condition just
57    // means an underestimation of the cost.
58    debug_assert!(
59        mem::size_of::<T>() as u64 <= T::DECLARED_SIZE,
60        "mem size: {}, declared: {}",
61        std::mem::size_of::<T>(),
62        T::DECLARED_SIZE
63    );
64    budget.as_budget().charge(
65        ContractCostType::MemCpy,
66        Some(n_elts.saturating_mul(T::DECLARED_SIZE)),
67    )
68}

70// Let it be a free function instead of a trait because charge_heap_alloc maybe called elsewhere,
71// not just metered clone, e.g. Box::<T>::new().
72pub(crate) fn charge_heap_alloc<T: DeclaredSizeForMetering>(
73    n_elts: u64,
74    budget: impl AsBudget,
75) -> Result<(), HostError> {
76    // Here we make a runtime assertion that the type's size is below its promised element size for
77    // budget charging.
78    debug_assert!(
79        mem::size_of::<T>() as u64 <= T::DECLARED_SIZE,
80        "mem size: {}, declared: {}",
81        std::mem::size_of::<T>(),
82        T::DECLARED_SIZE
83    );
84    budget.as_budget().charge(
85        ContractCostType::MemAlloc,
86        Some(n_elts.saturating_mul(T::DECLARED_SIZE)),
87    )
88}

101/// Represents a collection type which can be created from an iterator, and provides
102/// a method for bulk charging its creation cost. This does not clone elements from
103/// the iterator, just moves them into the container, so only charges shallow copies.
104pub trait MeteredContainer: FromIterator<Self::Item> + DeclaredSizeForMetering
105where
106    Self: Sized,
107{
108    type Item: DeclaredSizeForMetering;
109    /// Charges the cost for bulk initializing the container and copying `len` elements
110    /// into it.
111    fn charge_bulk_init_cpy(len: u64, budget: impl AsBudget) -> Result<(), HostError> {
112        charge_shallow_copy::<Self>(1, budget.clone())?;
113        charge_heap_alloc::<Self::Item>(len, budget.clone())?;
114        charge_shallow_copy::<Self::Item>(len, budget)
115    }
116
117    /// Metered version of the container's `with_capacity` method.
118    fn with_metered_capacity(capacity: usize, budget: impl AsBudget) -> Result<Self, HostError>;
119}
120
121impl<T: DeclaredSizeForMetering> MeteredContainer for Vec<T> {
122    type Item = T;
123
124    fn with_metered_capacity(capacity: usize, budget: impl AsBudget) -> Result<Self, HostError> {
125        Self::charge_bulk_init_cpy(capacity as u64, budget)?;
126        Ok(Vec::with_capacity(capacity))
127    }
128}

144/// Represents an iterator which can collect its elements into a `MeteredContainer`
145/// and automatically account for the metering cost.
146///
147/// Returns:
148///   Err - if either the budget was exceeded or the iterator lacks the size hint
149///   Ok - otherwise
150pub trait MeteredIterator: Iterator
151where
152    Self::Item: DeclaredSizeForMetering,
153    Self: Sized,
154{
155    fn metered_collect<B: MeteredContainer<Item = Self::Item>>(
156        self,
157        budget: impl AsBudget,
158    ) -> Result<B, HostError> {
159        if let (_, Some(ub)) = self.size_hint() {
160            B::charge_bulk_init_cpy(ub as u64, budget)?;
161            Ok(self.collect())
162        } else {
163            Err((ScErrorType::Object, ScErrorCode::InternalError).into())
164        }
165    }
166}

181/// A type that can be cloned, and charges for the cost of doing so, possibly
182/// failing if the clone runs out of budget. The cost of cloning may include
183/// both "shallow" and "substructure" components.
184pub trait MeteredClone: Clone + DeclaredSizeForMetering {
185    /// By default every MeteredClone type just charges as though it's shallow;
186    /// if a type is non-shallow (owns any substructure in the heap) it should
187    /// override both `IS_SHALLOW` (setting it to `false`) and
188    /// `charge_for_substructure` (correctly charging for cloning the
189    /// substructure).
190    const IS_SHALLOW: bool = true;
191
192    /// Called to clone the substructures of Self. The default implementation is
193    /// a no-op and is _only_ appropriate when Self is a shallow type (with no
194    /// substructure). If you override `Self::IS_SHALLOW` and set it to false,
195    /// you should override this method also (it will actually Err if you
196    /// don't). This charge does not include shallow copying of `Self` because
197    /// that should be taken care of by the caller beforehand, e.g. in
198    /// `metered_clone`.
199    fn charge_for_substructure(&self, _budget: impl AsBudget) -> Result<(), HostError> {
200        if Self::IS_SHALLOW {
201            Ok(())
202        } else {
203            Err((ScErrorType::Object, ScErrorCode::InternalError).into())
204        }
205    }
206
207    /// Called when cloning the substructures of a slice of Self. When
208    /// Self::IS_SHALLOW, this is a no-op. If Self::IS_SHALLOW is false, this
209    /// method will do element-wise charging for substructure, assuming it is
210    /// paired with element-wise cloning.
211    fn bulk_charge_for_substructure(
212        clones: &[Self],
213        budget: impl AsBudget,
214    ) -> Result<(), HostError> {
215        if Self::IS_SHALLOW {
216            // If we're shallow, just return.
217            Ok(())
218        } else {
219            for elt in clones {
220                elt.charge_for_substructure(budget.clone())?;
221            }
222            Ok(())
223        }
224    }
225
226    /// Convenience method for charging for a deep clone knowing the type is not
227    /// shallow.
228    fn charge_deep_clone(&self, budget: impl AsBudget) -> Result<(), HostError> {
229        if Self::IS_SHALLOW {
230            Err((ScErrorType::Object, ScErrorCode::InternalError).into())
231        } else {
232            charge_shallow_copy::<Self>(1, budget.clone())?;
233            self.charge_for_substructure(budget)
234        }
235    }
236
237    /// Composite helper handles metering and then, if successful, cloning. It
238    /// first charges for the shallow footprint of the type, followed by
239    /// charging the substructure, and finally calling `Clone::clone`. The
240    /// reason `charge_for_substructure` does not include shallow copying of
241    /// `Self` is because it is taken care of here. Typically overriding
242    /// `IS_SHALLOW` and `charge_for_substructure` should propertly take care of
243    /// custom non-trivial cloning. Thus this method should not need to be
244    /// overriden.
245    fn metered_clone(&self, budget: impl AsBudget) -> Result<Self, HostError> {
246        charge_shallow_copy::<Self>(1, budget.clone())?;
247        self.charge_for_substructure(budget)?;
248        Ok(self.clone())
249    }
250}

382    fn charge_for_substructure(&self, budget: impl AsBudget) -> Result<(), HostError> {

422impl MeteredClone for ScVal {
423    const IS_SHALLOW: bool = false;
424
425    fn charge_for_substructure(&self, budget: impl AsBudget) -> Result<(), HostError> {
426        // This is the depth limit checkpoint for `ScVal` cloning.
427        let mut budget_clone = budget.as_budget().clone();
428        budget_clone.with_limited_depth(|_| {
429            match self {
430                ScVal::Vec(Some(v)) => ScVec::charge_for_substructure(v, budget),
431                ScVal::Map(Some(m)) => ScMap::charge_for_substructure(m, budget),
432                ScVal::Vec(None) | ScVal::Map(None) => {
433                    Err((ScErrorType::Value, ScErrorCode::InvalidInput).into())
434                }
435                ScVal::Bytes(b) => BytesM::charge_for_substructure(b, budget),
436                ScVal::String(s) => StringM::charge_for_substructure(s, budget),
437                ScVal::Symbol(s) => StringM::charge_for_substructure(s, budget),
438                ScVal::ContractInstance(i) => {
439                    ScContractInstance::charge_for_substructure(i, budget)
440                }
441                // Everything else was handled by the memcpy above.
442                ScVal::U64(_)
443                | ScVal::I64(_)
444                | ScVal::U128(_)
445                | ScVal::I128(_)
446                | ScVal::Address(_)
447                | ScVal::U32(_)
448                | ScVal::I32(_)
449                | ScVal::Error(_)
450                | ScVal::Bool(_)
451                | ScVal::Void
452                | ScVal::Timepoint(_)
453                | ScVal::Duration(_)
454                | ScVal::U256(_)
455                | ScVal::I256(_)
456                | ScVal::LedgerKeyContractInstance
457                | ScVal::LedgerKeyNonce(_) => Ok(()),
458            }
459        })
460    }
461}
